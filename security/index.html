fun sha256(input: String): String {
    val digest = MessageDigest.getInstance("SHA-256")
    val hash   = digest.digest(input.toByteArray(Charsets.UTF_8))
    return hash.joinToString("") { "%02x".format(it) }
}

fun sha512(input: String): String =
    MessageDigest.getInstance("SHA-512")
        .digest(input.toByteArray())
        .joinToString("") { "%02x".format(it) }

// HMAC-SHA256
fun hmacSha256(data: String, key: String): String {
    val mac = Mac.getInstance("HmacSHA256")
    mac.init(SecretKeySpec(key.toByteArray(), "HmacSHA256"))
    return Base64.getEncoder().encodeToString(mac.doFinal(data.toByteArray()))
}

// Argon2id — современный стандарт (победитель PHC 2015)
type Argon2Params struct {
    Memory      uint32 // KB
    Iterations  uint32
    Parallelism uint8
    SaltLen     uint32
    KeyLen      uint32
}

// OWASP 2026: memory=64MB, iterations=3
var DefaultArgon2Params = Argon2Params{
    Memory:      64 * 1024, // 64 MB
    Iterations:  3,
    Parallelism: 4,
    SaltLen:     16,
    KeyLen:      32,
}

func hashPasswordArgon2(password string, p Argon2Params) (string, error) {
    salt := make([]byte, p.SaltLen)
    if _, err := io.ReadFull(rand.Reader, salt); err != nil {
        return "", fmt.Errorf("generate salt: %w", err)
    }

    hash := argon2.IDKey([]byte(password), salt,
        p.Iterations, p.Memory, p.Parallelism, p.KeyLen)

    // Формат: $argon2id$v=19$m=65536,t=3,p=4$<salt>$<hash>
    encoded := fmt.Sprintf("$argon2id$v=%d$m=%d,t=%d,p=%d$%s$%s",
        argon2.Version,
        p.Memory, p.Iterations, p.Parallelism,
        base64.RawStdEncoding.EncodeToString(salt),
        base64.RawStdEncoding.EncodeToString(hash),
    )
    return encoded, nil
}

func verifyPasswordArgon2(password, encodedHash string) (bool, error) {
    var version int
    var p Argon2Params
    var saltB64, hashB64 string

    _, err := fmt.Sscanf(encodedHash,
        "$argon2id$v=%d$m=%d,t=%d,p=%d$%s$%s",
        &version, &p.Memory, &p.Iterations, &p.Parallelism, &saltB64, &hashB64)
    if err != nil { return false, fmt.Errorf("decode hash: %w", err) }

    salt, _ := base64.RawStdEncoding.DecodeString(saltB64)
    expectedHash, _ := base64.RawStdEncoding.DecodeString(hashB64)
    p.KeyLen = uint32(len(expectedHash))

    actualHash := argon2.IDKey([]byte(password), salt,
        p.Iterations, p.Memory, p.Parallelism, p.KeyLen)

    // subtle.ConstantTimeCompare — защита от timing attacks
    return subtle.ConstantTimeCompare(actualHash, expectedHash) == 1, nil
}

    mac.Write(data)
    return mac.Sum(nil)

    import javax.crypto.*
import javax.crypto.spec.GCMParameterSpec

object AesGcm {
    private const val ALGORITHM = "AES/GCM/NoPadding"
    private const val KEY_SIZE  = 256
    private const val NONCE_LEN = 12           // 96 bit — рекомендовано
    private const val TAG_LEN   = 128          // bits

    fun generateKey(): SecretKey =
        KeyGenerator.getInstance("AES").apply { init(KEY_SIZE) }.generateKey()

    fun encrypt(plaintext: ByteArray, key: SecretKey): ByteArray {
        val nonce  = ByteArray(NONCE_LEN).also {
            java.security.SecureRandom().nextBytes(it)
        }
        val cipher = Cipher.getInstance(ALGORITHM).apply {
            init(Cipher.ENCRYPT_MODE, key, GCMParameterSpec(TAG_LEN, nonce))
        }
        val ciphertext = cipher.doFinal(plaintext)
        return nonce + ciphertext              // nonce || ciphertext+tag
    }

    fun decrypt(combined: ByteArray, key: SecretKey): ByteArray {
        val nonce      = combined.copyOfRange(0, NONCE_LEN)
        val ciphertext = combined.copyOfRange(NONCE_LEN, combined.size)
        return Cipher.getInstance(ALGORITHM).run {
            init(Cipher.DECRYPT_MODE, key, GCMParameterSpec(TAG_LEN, nonce))
            doFinal(ciphertext)
        }
    }
}

    // ③ AES-GCM — симметричное шифрование
// GCM = Galois/Counter Mode: authenticated encryption (шифрование + MAC)
// ВАЖНО: nonce никогда не повторять с тем же ключом!
// ──────────────────────────────────────────────
func encrypt(plaintext, key []byte) ([]byte, error) {
    if len(key) != 16 && len(key) != 24 && len(key) != 32 {
        return nil, errors.New("key must be 16, 24, or 32 bytes (AES-128/192/256)")
    }

    block, err := aes.NewCipher(key)
    if err != nil { return nil, err }

    gcm, err := cipher.NewGCM(block)
    if err != nil { return nil, err }

    // Nonce: 12 байт, криптографически случайный
    nonce := make([]byte, gcm.NonceSize())    // NonceSize() = 12
    if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
        return nil, fmt.Errorf("generate nonce: %w", err)
    }

    // Seal: шифрует И добавляет authentication tag (16 байт в конце)
    // Формат: [nonce(12) || ciphertext || tag(16)]
    ciphertext := gcm.Seal(nonce, nonce, plaintext, nil)
    return ciphertext, nil
}

func decrypt(ciphertext, key []byte) ([]byte, error) {
    block, err := aes.NewCipher(key)
    if err != nil { return nil, err }

    gcm, err := cipher.NewGCM(block)
    if err != nil { return nil, err }

    nonceSize := gcm.NonceSize()
    if len(ciphertext) < nonceSize {
        return nil, errors.New("ciphertext too short")
    }

    nonce, ciphertext := ciphertext[:nonceSize], ciphertext[nonceSize:]

    // Open: расшифровывает И проверяет authentication tag
    plaintext, err := gcm.Open(nil, nonce, ciphertext, nil)
    if err != nil {
        // Не раскрывай детали — только "decryption failed"
        return nil, errors.New("decryption failed")
    }
    return plaintext, nil
}

// Ключ из пароля через PBKDF2/Argon2
func deriveKey(password, salt []byte) []byte {
    // Argon2id → 32-байтный ключ для AES-256
    return argon2.IDKey(password, salt, 3, 64*1024, 4, 32)
}

func encryptWithPassword(plaintext []byte, password string) ([]byte, error) {
    salt := make([]byte, 16)
    if _, err := io.ReadFull(rand.Reader, salt); err != nil { return nil, err }

    key := deriveKey([]byte(password), salt)
    encrypted, err := encrypt(plaintext, key)
    if err != nil { return nil, err }

    // Формат: [salt(16) || encrypted_data]
    result := make([]byte, len(salt)+len(encrypted))
    copy(result, salt)
    copy(result[len(salt):], encrypted)
    return result, nil
}

func decryptWithPassword(data []byte, password string) ([]byte, error) {
    if len(data) < 16 { return nil, errors.New("data too short") }
    salt, encrypted := data[:16], data[16:]
    key := deriveKey([]byte(password), salt)
    return decrypt(encrypted, key)
}

type Claims struct {

type JWTManager struct {
    signingKey     []byte         // для HS256
    accessExpiry   time.Duration
    refreshExpiry  time.Duration
    issuer         string
}

func NewJWTManager(secret string) *JWTManager {
    return &JWTManager{
        signingKey:    []byte(secret),
        accessExpiry:  15 * time.Minute,  // короткий срок!
        refreshExpiry: 7 * 24 * time.Hour,
        issuer:        "myapp",
    }
}

func (m *JWTManager) GenerateAccessToken(userID, email, role string) (string, error) {
    claims := Claims{
        UserID: userID,
        Email:  email,
        Role:   role,
        RegisteredClaims: jwt.RegisteredClaims{
            Issuer:    m.issuer,
            Subject:   userID,
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(m.accessExpiry)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
            NotBefore: jwt.NewNumericDate(time.Now()),
            ID:        generateTokenID(),           // jti: уникальный ID для отзыва
        },
    }
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString(m.signingKey)
}

func (m *JWTManager) Verify(tokenString string) (*Claims, error) {
    token, err := jwt.ParseWithClaims(
        tokenString,
        &Claims{},
        func(t *jwt.Token) (any, error) {
            // ОБЯЗАТЕЛЬНО проверить алгоритм — защита от alg=none атаки
            if _, ok := t.Method.(*jwt.SigningMethodHMAC); !ok {
                return nil, fmt.Errorf("unexpected signing method: %v", t.Header["alg"])
            }
            return m.signingKey, nil
        },
        jwt.WithIssuer(m.issuer),
        jwt.WithExpirationRequired(),          // require exp claim
        jwt.WithValidMethods([]string{"HS256"}), // whitelist алгоритмов
    )

    if err != nil {
        switch {
        case errors.Is(err, jwt.ErrTokenExpired):
            return nil, ErrTokenExpired
        case errors.Is(err, jwt.ErrTokenSignatureInvalid):
            return nil, ErrTokenInvalid
        default:
            return nil, fmt.Errorf("parse token: %w", err)
        }
    }

    claims, ok := token.Claims.(*Claims)
    if !ok || !token.Valid {
        return nil, ErrTokenInvalid
    }
    return claims, nil
}

// JWT Middleware для net/http
func (m *JWTManager) Middleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        authHeader := r.Header.Get("Authorization")
        if authHeader == "" || !strings.HasPrefix(authHeader, "Bearer ") {
            http.Error(w, `{"error":"missing authorization header"}`,
                http.StatusUnauthorized)
            return
        }

        tokenString := strings.TrimPrefix(authHeader, "Bearer ")
        claims, err := m.Verify(tokenString)
        if err != nil {
            http.Error(w, fmt.Sprintf(`{"error":%q}`, err.Error()),
                http.StatusUnauthorized)
            return
        }

        // Добавить claims в context
        ctx := context.WithValue(r.Context(), claimsKey, claims)
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

func secureRandom() {
    // Токен сессии
    token := make([]byte, 32)
    io.ReadFull(rand.Reader, token)
    sessionToken := base64.URLEncoding.EncodeToString(token) // URL-safe

    // Случайная строка (для CSRF токенов, API keys)
    apiKey := make([]byte, 24)
    io.ReadFull(rand.Reader, apiKey)
    key := base64.RawURLEncoding.EncodeToString(apiKey)

    // Случайное число в диапазоне [0, max)
    import "math/big"
    max := big.NewInt(1_000_000)
    n, _ := rand.Int(rand.Reader, max)        // cryptographically secure!
    otp := n.Int64()                          // OTP-код

    _ = sessionToken; _ = key; _ = otp
}

// ⑥ constant-time сравнение — против timing attacks
// ──────────────────────────────────────────────
func safeCompare(a, b string) bool {
    // subtle.ConstantTimeCompare — одинаковое время вне зависимости от совпадения
    return subtle.ConstantTimeCompare([]byte(a), []byte(b)) == 1
}

    // НИКОГДА не используй == для сравнения секретов!
    // if provided == expected { ... }        // УЯЗВИМО к timing attack
    return safeCompare(provided, expected)    // БЕЗОПАСНО
}

    Безопасное хранение секретов (Keychain / KeyStore / keyring)🔗

    import Security

enum KeychainError: Error {
    case duplicateItem, itemNotFound, invalidData
    case unexpectedStatus(OSStatus)
}

struct KeychainManager {
    static let service = Bundle.main.bundleIdentifier ?? "com.myapp"

    // Сохранить (создать или обновить)
    static func save(_ value: String, for key: String) throws {
        guard let data = value.data(using: .utf8) else {
            throw KeychainError.invalidData
        }
        let query: [CFString: Any] = [
            kSecClass:       kSecClassGenericPassword,
            kSecAttrService: service,
            kSecAttrAccount: key,
        ]
        let update: [CFString: Any] = [kSecValueData: data]
        let status = SecItemUpdate(query as CFDictionary, update as CFDictionary)

        if status == errSecItemNotFound {
            var addQuery = query
            addQuery[kSecValueData] = data
            addQuery[kSecAttrAccessible] = kSecAttrAccessibleWhenUnlockedThisDeviceOnly
            let addStatus = SecItemAdd(addQuery as CFDictionary, nil)
            guard addStatus == errSecSuccess else {
                throw KeychainError.unexpectedStatus(addStatus)
            }
        } else if status != errSecSuccess {
            throw KeychainError.unexpectedStatus(status)
        }
    }

    // Прочитать
    static func load(key: String) throws -> String {
        let query: [CFString: Any] = [
            kSecClass:            kSecClassGenericPassword,
            kSecAttrService:      service,
            kSecAttrAccount:      key,
            kSecReturnData:       true,
            kSecMatchLimit:       kSecMatchLimitOne,
        ]
        var result: CFTypeRef?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess, let data = result as? Data,
              let string = String(data: data, encoding: .utf8) else {
            throw status == errSecItemNotFound
                ? KeychainError.itemNotFound
                : KeychainError.unexpectedStatus(status)
        }
        return string
    }

    // Удалить
    static func delete(key: String) throws {
        let query: [CFString: Any] = [
            kSecClass:       kSecClassGenericPassword,
            kSecAttrService: service,
            kSecAttrAccount: key,
        ]
        let status = SecItemDelete(query as CFDictionary)
        guard status == errSecSuccess || status == errSecItemNotFound else {
            throw KeychainError.unexpectedStatus(status)
        }
    }
}

import android.security.keystore.KeyProperties
import java.security.KeyStore

object AndroidKeystore {
    private const val PROVIDER  = "AndroidKeyStore"
    private const val ALIAS     = "MyAppKey"
    private const val ALGORITHM = "AES/GCM/NoPadding"

    private fun getOrCreateKey(): SecretKey {
        val keyStore = KeyStore.getInstance(PROVIDER).apply { load(null) }
        keyStore.getKey(ALIAS, null)?.let { return it as SecretKey }

        val spec = KeyGenParameterSpec.Builder(
            ALIAS,
            KeyProperties.PURPOSE_ENCRYPT or KeyProperties.PURPOSE_DECRYPT
        )
            .setBlockModes(KeyProperties.BLOCK_MODE_GCM)
            .setEncryptionPaddings(KeyProperties.ENCRYPTION_PADDING_NONE)
            .setKeySize(256)
            .setUserAuthenticationRequired(false) // true = биометрия
            .build()

        return KeyGenerator.getInstance(
            KeyProperties.KEY_ALGORITHM_AES, PROVIDER
        ).apply { init(spec) }.generateKey()
    }

    fun encrypt(data: String): Pair<ByteArray, ByteArray> {
        val key    = getOrCreateKey()
        val cipher = Cipher.getInstance(ALGORITHM).apply {
            init(Cipher.ENCRYPT_MODE, key)
        }
        return Pair(cipher.iv, cipher.doFinal(data.toByteArray()))
    }

    fun decrypt(iv: ByteArray, ciphertext: ByteArray): String {
        val key = getOrCreateKey()
        val cipher = Cipher.getInstance(ALGORITHM).apply {
            init(Cipher.DECRYPT_MODE, key, GCMParameterSpec(128, iv))
        }
        return String(cipher.doFinal(ciphertext))
    }
}

import JWTKit

struct UserPayload: JWTPayload {
    var sub: SubjectClaim
    var exp: ExpirationClaim
    var iat: IssuedAtClaim
    var role: String

    func verify(using key: some JWTAlgorithm) async throws {
        try exp.verifyNotExpired()
    }
}

// Создание
let keyCollection = await JWTKeyCollection()
    .add(hmac: "secret-key-min-32-chars-long!!", digestAlgorithm: .sha256)

let payload = UserPayload(
    sub: "user-123",
    exp: .init(value: Date().addingTimeInterval(3600)),
    iat: .init(value: Date()),
    role: "admin"
)
let token = try await keyCollection.sign(payload)

// Верификация
let verified = try await keyCollection.verify(token, as: UserPayload.self)
print(verified.sub.value)                      // "user-123"

import io.jsonwebtoken.*
import io.jsonwebtoken.security.Keys
import java.util.Date

object JwtService {
    private val SECRET_KEY = Keys.hmacShaKeyFor(
        "your-super-secret-key-at-least-32-chars!".toByteArray()
    )
    private const val EXPIRATION_MS = 3_600_000L  // 1 час

    fun generateToken(userId: String, role: String): String =
        Jwts.builder()
            .subject(userId)
            .claim("role", role)
            .issuedAt(Date())
            .expiration(Date(System.currentTimeMillis() + EXPIRATION_MS))
            .signWith(SECRET_KEY)
            .compact()

    fun validateToken(token: String): Claims =
        Jwts.parser()
            .verifyWith(SECRET_KEY)
            .build()
            .parseSignedClaims(token)
            .payload
}

val token = JwtService.generateToken("user-123", "admin")
val claims = JwtService.validateToken(token)
println(claims.subject)                        // "user-123"
println(claims["role"])                         // "admin"

import jwt
import os, secrets
from datetime import datetime, timedelta, UTC

SECRET_KEY = os.environ.get("JWT_SECRET", secrets.token_hex(32))

def create_token(
    user_id: str, role: str,
    expires_in: timedelta = timedelta(hours=1)
) -> str:
    payload = {
        "sub":  user_id,
        "role": role,
        "iat":  datetime.now(UTC),
        "exp":  datetime.now(UTC) + expires_in,
    }
    return jwt.encode(payload, SECRET_KEY, algorithm="HS256")

def verify_token(token: str) -> dict:
    try:
        return jwt.decode(
            token, SECRET_KEY,
            algorithms=["HS256"],
            options={"require": ["sub", "exp", "iat"]}
        )
    except jwt.ExpiredSignatureError:
        raise ValueError("Token has expired")
    except jwt.InvalidTokenError as e:
        raise ValueError(f"Invalid token: {e}")

)

# Использование в FastAPI
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

security = HTTPBearer()

async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security)
) -> dict:
    try:
        return verify_token(credentials.credentials)
    except ValueError as e:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail=str(e),
            headers={"WWW-Authenticate": "Bearer"},
        )
