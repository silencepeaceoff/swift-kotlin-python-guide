    Динамический JSON¶

    import Foundation

// JSONSerialization — когда структура JSON неизвестна заранее
let jsonData = """
{"user": {"name": "Alice", "age": 30, "tags": ["admin", "user"]}}
""".data(using: .utf8)!

let raw = try JSONSerialization.jsonObject(with: jsonData)
if let dict = raw as? [String: Any],
   let user = dict["user"] as? [String: Any],
   let name = user["name"] as? String,
   let age  = user["age"]  as? Int {
    print("\(name), \(age)")
}

// Type-safe JSON enum для произвольных структур
indirect enum JSON: Codable {
    case null
    case bool(Bool)
    case int(Int)
    case double(Double)
    case string(String)
    case array([JSON])
    case object([String: JSON])

    subscript(key: String) -> JSON? {
        if case .object(let dict) = self { return dict[key] }
        return nil
    }
    subscript(index: Int) -> JSON? {
        if case .array(let arr) = self, arr.indices.contains(index) {
            return arr[index]
        }
        return nil
    }
}

    import json
from datetime import datetime

# Кастомный encoder для нестандартных типов
class AppEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        if hasattr(obj, "__dict__"):
            return obj.__dict__
        return super().default(obj)

data = {"ts": datetime.now(), "value": 42}
json_str = json.dumps(data, cls=AppEncoder, indent=2, ensure_ascii=False)

    XML¶

    import Foundation

// XMLParser — SAX-парсер (event-based, встроен)
class SimpleXMLParser: NSObject, XMLParserDelegate {
    var result: [String: String] = [:]
    private var currentElement = ""
    private var currentValue   = ""

    func parser(_ parser: XMLParser,
                didStartElement element: String,
                namespaceURI: String?,
                qualifiedName: String?,
                attributes: [String: String]) {
        currentElement = element
        currentValue   = ""
    }

    func parser(_ parser: XMLParser, foundCharacters string: String) {
        currentValue += string
    }

    func parser(_ parser: XMLParser,
                didEndElement element: String,
                namespaceURI: String?,
                qualifiedName: String?) {
        result[element] = currentValue.trimmingCharacters(in: .whitespaces)
    }
}

let xml = """
<user><id>1</id><name>Alice</name><email>alice@example.com</email></user>
""".data(using: .utf8)!

let delegate = SimpleXMLParser()
let xmlParser = XMLParser(data: xml)
xmlParser.delegate = delegate
xmlParser.parse()
print(delegate.result)                         // ["id": "1", "name": "Alice", ...]

    import xml.etree.ElementTree as ET

xml_str = """<?xml version="1.0" encoding="utf-8"?>
<users>
    <user id="1" role="admin">
        <name>Alice</name>
        <email>alice@example.com</email>
    </user>
    <user id="2" role="user">
        <name>Bob</name>
        <email>bob@example.com</email>
    </user>
</users>"""

root = ET.fromstring(xml_str)

# Итерация
for user in root.findall("user"):
    uid   = user.attrib["id"]
    role  = user.attrib["role"]
    name  = user.findtext("name")
    email = user.findtext("email")
    print(f"[{uid}] {name} ({role}): {email}")

# XPath поиск
admins = root.findall(".//user[@role='admin']")
alice  = root.find(".//user[name='Alice']")

# Создание XML

ET.SubElement(user_elem, "name").text = "Charlie"

tree = ET.ElementTree(new_root)

tree.write("users.xml", encoding="utf-8", xml_declaration=True)

// Decode в Codable
struct AppConfig: Codable {
    struct Server: Codable { let host: String; let port: Int }
    struct Database: Codable { let url: String; let poolSize: Int }
    let server: Server
    let database: Database
    let features: [String]
}

let config = try YAMLDecoder().decode(AppConfig.self, from: yamlString)
print(config.server.host)                      // "localhost"
print(config.features)                         // ["auth", "notifications"]

// Encode → YAML
let encoded = try YAMLEncoder().encode(config)

# Преимущество: YAML 1.2, сохранение комментариев, roundtrip
from ruamel.yaml import YAML
from io import StringIO

yaml = YAML()
yaml.preserve_quotes = True

config_str = """
# Конфигурация сервера
server:
  host: localhost
  port: 8080           # порт по умолчанию
database:
  url: postgresql://localhost/mydb
  pool_size: 10
features:
  - auth
  - notifications
"""

# Загрузка (CommentedMap — сохраняет комментарии!)
data = yaml.load(StringIO(config_str))
print(data["server"]["port"])                  # 8080

# Изменение + сохранение (комментарии сохранены!)
data["server"]["port"] = 9090
with open("config.yaml", "w") as f:
    yaml.dump(data, f)

    import csv
from dataclasses import dataclass

@dataclass
class Person:
    name: str
    age: int
    city: str

# Запись
people = [Person("Alice", 30, "Berlin"), Person("Bob", 25, "Munich")]
with open("data.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["name", "age", "city"])
    writer.writeheader()
    for p in people:
        writer.writerow({"name": p.name, "age": p.age, "city": p.city})

# Чтение с type conversion
def read_people(path: str) -> list[Person]:
    with open(path, encoding="utf-8") as f:
        return [
            Person(name=row["name"], age=int(row["age"]), city=row["city"])
            for row in csv.DictReader(f)
        ]

# Ленивое чтение больших файлов — генератор
def iter_csv(path: str):
    with open(path, encoding="utf-8") as f:
        yield from csv.DictReader(f)

for row in iter_csv("huge.csv"):
