    Хранение в памяти🔗

    Концепция Swift Kotlin Python
    Value types struct, enum, базовые типы (стек / inline в куче) Int, Boolean и др. компилируются как JVM-примитивы — (всё является объектом)
    Reference types class (ARC на куче) class, все коллекции (JVM heap) Все объекты (int, str, list — на куче)
    Управление памятью ARC (Automatic Reference Counting) — детерминированно JVM GC (G1 / ZGC / Shenandoah) CPython: подсчёт ссылок + циклический GC (gc модуль)
    Null-safety Compile-time (Optional) Compile-time (?) Runtime (None)
    Копирование Value types копируются при присваивании (COW-оптимизация) Объекты передаются по ссылке Все объекты — по ссылке; числа/строки — immutable

    CPython Integer Caching

    CPython кэширует объекты int в диапазоне −5 … 256 и интернирует короткие строки-идентификаторы. Это означает, что a = 256; b = 256; a is b вернёт True (один объект), а a = 257; b = 257; a is b — False (два разных объекта). Для сравнения значений всегда используй ==, а не is.

    package main

import (
    "fmt"
    "math"
    "math/big"
    "math/bits"
)

func numbers() {
    // ──────────────────────────────────────────
    // Целые числа
    // ──────────────────────────────────────────
    var a int   = 100
    var b int64 = 9_223_372_036_854_775_807  // разделитель _ для читаемости (Go 1.13+)
    var c uint  = 0xFF                        // hex литерал
    var d int   = 0b1010_1010                 // binary литерал (Go 1.13+)
    var e int   = 0o755                       // octal литерал (Go 1.13+)
    _ = a; _ = b; _ = c; _ = d; _ = e

    // Целочисленная арифметика
    fmt.Println(10 / 3)                       // 3 (truncated)
    fmt.Println(10 % 3)                       // 1
    fmt.Println(-7 % 3)                       // -1 (знак = знак делимого)

    // ──────────────────────────────────────────
    // Числа с плавающей точкой
    // ──────────────────────────────────────────
    x := 3.14
    y := 2.71828
    fmt.Printf("%.4f\n", x + y)              // форматирование

    // Специальные значения
    posInf := math.Inf(1)                     // +∞
    negInf := math.Inf(-1)                    // -∞
    nan    := math.NaN()
    fmt.Println(math.IsInf(posInf, 1))        // true
    fmt.Println(math.IsNaN(nan))              // true

    // Сравнение float — ТОЛЬКО через epsilon!
    const eps = 1e-9
    func floatEqual(a, b float64) bool { return math.Abs(a-b) < eps }

    // ──────────────────────────────────────────
    // Математические функции
    // ──────────────────────────────────────────
    math.Abs(-5.0)                            // 5.0
    math.Sqrt(16.0)                           // 4.0
    math.Pow(2, 10)                           // 1024.0
    math.Log(math.E)                          // 1.0
    math.Log2(1024)                           // 10.0
    math.Log10(1000)                          // 3.0
    math.Floor(3.9)                           // 3.0
    math.Ceil(3.1)                            // 4.0
    math.Round(3.5)                           // 4.0 (банковское: Round(2.5) = 2)
    math.Max(3.0, 5.0)                        // 5.0
    math.Min(3.0, 5.0)                        // 3.0

    // Тригонометрия
    math.Sin(math.Pi / 2)                     // 1.0
    math.Cos(0)                               // 1.0

    // ──────────────────────────────────────────
    // Битовые операции
    // ──────────────────────────────────────────
    n := 0b1010
    fmt.Println(n & 0b0110)                   // AND: 0b0010 = 2
    fmt.Println(n | 0b0110)                   // OR:  0b1110 = 14
    fmt.Println(n ^ 0b0110)                   // XOR: 0b1100 = 12
    fmt.Println(n << 2)                       // left shift: 40
    fmt.Println(n >> 1)                       // right shift: 5
    fmt.Println(^n)                           // NOT: -11 (bitwise complement)
    fmt.Println(n &^ 0b0110)                  // AND NOT (bit clear): 0b1000 = 8

    // math/bits — подсчёт бит (Go 1.9+)
    bits.OnesCount(0b1010_1010)               // 4 (количество единиц)
    bits.Len(255)                             // 8 (длина в битах)
    bits.LeadingZeros(1)                      // 63 (для uint64)

    // ──────────────────────────────────────────
    // big.Int — произвольная точность
    // ──────────────────────────────────────────
    a_big := new(big.Int).SetInt64(1_000_000_000_000)
    b_big := new(big.Int).SetInt64(999_999_999_999)
    sum   := new(big.Int).Add(a_big, b_big)
    fmt.Println(sum.String())                 // "1999999999999"

    // Факториал 100
    result := new(big.Int).SetInt64(1)
    for i := int64(2); i <= 100; i++ {
        result.Mul(result, big.NewInt(i))
    }
    fmt.Println(result.String())              // 93326215443944152681699238856266...

    // big.Float — произвольная точность float
    pi, _, _ := big.ParseFloat("3.14159265358979323846", 10, 256, big.ToNearestEven)
    fmt.Println(pi.Text('f', 30))             // 30 знаков после запятой
}

import (

func stringBasics() {
    // ──────────────────────────────────────────
    // Литералы
    // ──────────────────────────────────────────
    s  := "Hello, 世界"           // UTF-8 по умолчанию
    raw := `Multiline
raw string \n не экранируется`    // raw string literal (backticks)

    // Длина: байты vs руны!
    fmt.Println(len(s))                       // 13 байт (не символов!)
    fmt.Println(utf8.RuneCountInString(s))    // 9 рун (символов)

    // Итерация — ВСЕГДА используй range для Unicode
    for i, r := range s {
        fmt.Printf("pos=%d rune=%c (%d)\n", i, r, r)
    }

    // Явное преобразование string ↔ []rune (для манипуляций)
    runes := []rune(s)
    runes = 'h'
    modified := string(runes)                 // "hello, 世界"

    // string ↔ []byte (для бинарных операций)
    bytes := []byte(s)
    bytes = 'H'
    _ = string(bytes)                         // "Hello, 世界"
    _ = raw; _ = modified

    // ──────────────────────────────────────────
    // Форматирование (fmt.Sprintf)
    // ──────────────────────────────────────────
    name, age, score := "Alice", 30, 98.5
    fmt.Sprintf("Name: %s, Age: %d, Score: %.2f", name, age, score)
    fmt.Sprintf("Binary: %b, Hex: %x, Oct: %o", 42, 255, 8)
    fmt.Sprintf("Width: %10s, %-10s", "right", "left")   // выравнивание
    fmt.Sprintf("Padded: %05d", 42)           // "00042"
    fmt.Sprintf("Pointer: %p", &name)
    fmt.Sprintf("Type: %T, Value: %v", name, name)  // "string", "Alice"
    fmt.Sprintf("Struct: %+v", struct{ X, Y int }{1, 2})  // {X:1 Y:2}

    // ──────────────────────────────────────────
    // strings package — основные операции
    // ──────────────────────────────────────────
    s2 := "  Hello, World!  "

    strings.TrimSpace(s2)                     // "Hello, World!"
    strings.Trim(s2, " !")                    // "Hello, World"
    strings.TrimLeft(s2, " ")                 // "Hello, World!  "
    strings.TrimRight(s2, " ")                // "  Hello, World!"
    strings.TrimPrefix("photo.jpg", "photo.") // "jpg"
    strings.TrimSuffix("photo.jpg", ".jpg")   // "photo"

    strings.ToUpper("hello")                  // "HELLO"
    strings.ToLower("HELLO")                  // "hello"
    strings.Title("hello world")              // "Hello World" (deprecated, use cases.Title)

    strings.HasPrefix("filename.go", "file")  // true
    strings.HasSuffix("filename.go", ".go")   // true
    strings.Contains("hello world", "world")  // true
    strings.ContainsAny("hello", "aeiou")     // true (есть хоть одна из букв)
    strings.Count("cheese", "e")             // 3
    strings.Index("hello", "ll")             // 2 (-1 если не найдено)
    strings.LastIndex("hello", "l")          // 3

    strings.Replace("oink oink oink", "oink", "moo", 2) // "moo moo oink"
    strings.ReplaceAll("foo bar foo", "foo", "baz")      // "baz bar baz"

    strings.Split("a,b,c", ",")              // ["a" "b" "c"]
    strings.SplitN("a,b,c", ",", 2)         // ["a" "b,c"]
    strings.Fields("  foo  bar  ")           // ["foo" "bar"] (split by whitespace)
    strings.Join([]string{"a", "b", "c"}, "-") // "a-b-c"

    strings.Repeat("Go!", 3)                 // "Go!Go!Go!"

    // ──────────────────────────────────────────
    // strings.Builder — эффективная конкатенация
    // ──────────────────────────────────────────
    var b strings.Builder
    for i := 0; i < 5; i++ {
        fmt.Fprintf(&b, "item%d ", i)
    }
    result := b.String()                      // "item0 item1 item2 item3 item4 "
    b.Reset()                                 // очистить для повторного использования
    _ = result

    // Конкатенация operator + (только для небольшого числа строк!)
    greeting := "Hello" + ", " + "World!"    // создаёт 2 промежуточных строки

    // ──────────────────────────────────────────
    // strconv — конвертация строк ↔ числа
    // ──────────────────────────────────────────
    import "strconv"
    n, err := strconv.Atoi("42")             // string → int
    s3 := strconv.Itoa(42)                   // int → string
    f, _ := strconv.ParseFloat("3.14", 64)   // string → float64
    fs := strconv.FormatFloat(3.14, 'f', 2, 64)  // float64 → string: "3.14"
    b2, _ := strconv.ParseBool("true")       // "true"/"false"/"1"/"0" → bool
    _ = n; _ = err; _ = s3; _ = f; _ = fs; _ = b2
    _ = greeting

    // Unicode проверки
    unicode.IsLetter('A')                     // true
    unicode.IsDigit('5')                      // true
    unicode.IsSpace(' ')                      // true
    unicode.IsUpper('A')                      // true
    unicode.ToLower('A')                      // 'a'
}
