    Memory Storage🔗

    Concept Swift Kotlin Python
    Value types struct, enum, basic types (stack / inline on heap) Int, Boolean etc. compile as JVM primitives — (everything is an object)
    Reference types class (ARC on heap) class, all collections (JVM heap) All objects (int, str, list — on heap)
    Memory Management ARC (Automatic Reference Counting) — deterministic JVM GC (G1 / ZGC / Shenandoah) CPython: Ref-counting + cyclic GC (gc module)
    Null-safety Compile-time (Optional) Compile-time (?) Runtime (None)
    Copy behavior Value types copy on assignment (COW optimization) Objects are passed by reference All objects by reference; numbers/strings — immutable

    CPython Integer Caching

    CPython caches int objects in the range −5 … 256 and interns short identifiers. This means a = 256; b = 256; a is b returns True (same object), whereas a = 257; b = 257; a is b returns False (two distinct objects). Always use == instead of is to compare values.

    package main

import (
    "fmt"
    "math"
    "math/big"
    "math/bits"
)

func numbers() {
    // ──────────────────────────────────────────
    // Integers
    // ──────────────────────────────────────────
    var a int   = 100
    var b int64 = 9_223_372_036_854_775_807  // separator _ for readability (Go 1.13+)
    var c uint  = 0xFF                        // hex literal
    var d int   = 0b1010_1010                 // binary literal (Go 1.13+)
    var e int   = 0o755                       // octal literal (Go 1.13+)
    _ = a; _ = b; _ = c; _ = d; _ = e

    // Integer arithmetic
    fmt.Println(10 / 3)                       // 3 (truncated)
    fmt.Println(10 % 3)                       // 1
    fmt.Println(-7 % 3)                       // -1 (sign matches dividend)

    // ──────────────────────────────────────────
    // Floats
    // ──────────────────────────────────────────
    x := 3.14
    y := 2.71828
    fmt.Printf("%.4f\n", x + y)              // formatting

    // Special values
    posInf := math.Inf(1)                     // +∞
    negInf := math.Inf(-1)                    // -∞
    nan    := math.NaN()
    fmt.Println(math.IsInf(posInf, 1))        // true
    fmt.Println(math.IsNaN(nan))              // true

    // Float comparison — ONLY via epsilon!
    const eps = 1e-9
    func floatEqual(a, b float64) bool { return math.Abs(a-b) < eps }

    // ──────────────────────────────────────────
    // Math functions
    // ──────────────────────────────────────────
    math.Abs(-5.0)                            // 5.0
    math.Sqrt(16.0)                           // 4.0
    math.Pow(2, 10)                           // 1024.0
    math.Log(math.E)                          // 1.0
    math.Log2(1024)                           // 10.0
    math.Log10(1000)                          // 3.0
    math.Floor(3.9)                           // 3.0
    math.Ceil(3.1)                            // 4.0
    math.Round(3.5)                           // 4.0 (Banker's: Round(2.5) = 2)
    math.Max(3.0, 5.0)                        // 5.0
    math.Min(3.0, 5.0)                        // 3.0

    // Trigonometry
    math.Sin(math.Pi / 2)                     // 1.0
    math.Cos(0)                               // 1.0

    // ──────────────────────────────────────────
    // Bitwise operations
    // ──────────────────────────────────────────
    n := 0b1010
    fmt.Println(n & 0b0110)                   // AND: 0b0010 = 2
    fmt.Println(n | 0b0110)                   // OR:  0b1110 = 14
    fmt.Println(n ^ 0b0110)                   // XOR: 0b1100 = 12
    fmt.Println(n << 2)                       // left shift: 40
    fmt.Println(n >> 1)                       // right shift: 5
    fmt.Println(^n)                           // NOT: -11 (bitwise complement)
    fmt.Println(n &^ 0b0110)                  // AND NOT (bit clear): 0b1000 = 8

    // math/bits — bit counting (Go 1.9+)
    bits.OnesCount(0b1010_1010)               // 4 (set bits)
    bits.Len(255)                             // 8 (bit length)
    bits.LeadingZeros(1)                      // 63 (for uint64)

    // ──────────────────────────────────────────
    // big.Int — arbitrary precision
    // ──────────────────────────────────────────
    a_big := new(big.Int).SetInt64(1_000_000_000_000)
    b_big := new(big.Int).SetInt64(999_999_999_999)
    sum   := new(big.Int).Add(a_big, b_big)
    fmt.Println(sum.String())                 // "1999999999999"

    // Factorial 100
    result := new(big.Int).SetInt64(1)
    for i := int64(2); i <= 100; i++ {
        result.Mul(result, big.NewInt(i))
    }
    fmt.Println(result.String())              // 93326215443944152681699238856266...

    // big.Float — arbitrary precision float
    pi, _, _ := big.ParseFloat("3.14159265358979323846", 10, 256, big.ToNearestEven)
    fmt.Println(pi.Text('f', 30))             // 30 decimal places
}

    package main

import (
    "fmt"
    "strings"
    "unicode"
    "unicode/utf8"
)

func stringBasics() {
    // ──────────────────────────────────────────
    // Literals
    // ──────────────────────────────────────────
    s  := "Hello, 世界"           // Default is UTF-8
    raw := `Multiline
raw string \n not escaped`        // raw string literal (backticks)

    // Length: bytes vs runes!
    fmt.Println(len(s))                       // 13 bytes (not characters!)
    fmt.Println(utf8.RuneCountInString(s))    // 9 runes (characters)

    // Iteration — ALWAYS use range for Unicode
    for i, r := range s {
        fmt.Printf("pos=%d rune=%c (%d)\n", i, r, r)
    }

    // Explicit conversion string ↔ []rune (for manipulations)
    runes := []rune(s)
    runes[0] = 'h'
    modified := string(runes)                 // "hello, 世界"

    // string ↔ []byte (for binary ops)
    bytes := []byte(s)
    bytes[0] = 'H'
    _ = string(bytes)                         // "Hello, 世界"
    _ = raw; _ = modified

    // ──────────────────────────────────────────
    // Formatting (fmt.Sprintf)
    // ──────────────────────────────────────────
    name, age, score := "Alice", 30, 98.5
    fmt.Sprintf("Name: %s, Age: %d, Score: %.2f", name, age, score)
    fmt.Sprintf("Binary: %b, Hex: %x, Oct: %o", 42, 255, 8)
    fmt.Sprintf("Width: %10s, %-10s", "right", "left")   // alignment
    fmt.Sprintf("Padded: %05d", 42)           // "00042"
    fmt.Sprintf("Pointer: %p", &name)
    fmt.Sprintf("Type: %T, Value: %v", name, name)  // "string", "Alice"
    fmt.Sprintf("Struct: %+v", struct{ X, Y int }{1, 2})  // {X:1 Y:2}

    // ──────────────────────────────────────────
    // strings package — basic operations
    // ──────────────────────────────────────────
    s2 := "  Hello, World!  "

    strings.TrimSpace(s2)                     // "Hello, World!"
    strings.Trim(s2, " !")                    // "Hello, World"
    strings.TrimLeft(s2, " ")                 // "Hello, World!  "
    strings.TrimRight(s2, " ")                // "  Hello, World!"
    strings.TrimPrefix("photo.jpg", "photo.") // "jpg"
    strings.TrimSuffix("photo.jpg", ".jpg")   // "photo"

    strings.ToUpper("hello")                  // "HELLO"
    strings.ToLower("HELLO")                  // "hello"
    // strings.Title is deprecated, use cases.Title instead

    strings.HasPrefix("filename.go", "file")  // true
    strings.HasSuffix("filename.go", ".go")   // true
    strings.Contains("hello world", "world")  // true
    strings.ContainsAny("hello", "aeiou")     // true (has at least one letter)
    strings.Count("cheese", "e")             // 3
    strings.Index("hello", "ll")             // 2 (-1 if not found)
    strings.LastIndex("hello", "l")          // 3

    strings.Replace("oink oink oink", "oink", "moo", 2) // "moo moo oink"
    strings.ReplaceAll("foo bar foo", "foo", "baz")      // "baz bar baz"

    strings.Split("a,b,c", ",")              // ["a", "b", "c"]
    strings.SplitN("a,b,c", ",", 2)         // ["a", "b,c"]
    strings.Fields("  foo  bar  ")           // ["foo", "bar"] (split by whitespace)
    strings.Join([]string{"a", "b", "c"}, "-") // "a-b-c"

    strings.Repeat("Go!", 3)                 // "Go!Go!Go!"

    // ──────────────────────────────────────────
    // strings.Builder — efficient concatenation
    // ──────────────────────────────────────────
    var b strings.Builder
    for i := 0; i < 5; i++ {
        fmt.Fprintf(&b, "item%d ", i)
    }
    result := b.String()                      // "item0 item1 item2 item3 item4 "
    b.Reset()                                 // clear for reuse
    _ = result

    // Operator + concatenation (only for short amounts of strings!)
    greeting := "Hello" + ", " + "World!"    // creates 2 intermediate strings

    // ──────────────────────────────────────────
    // strconv — conversions string ↔ numbers
    // ──────────────────────────────────────────
    import "strconv"
    n, err := strconv.Atoi("42")             // string → int
    s3 := strconv.Itoa(42)                   // int → string
    f, _ := strconv.ParseFloat("3.14", 64)   // string → float64
    fs := strconv.FormatFloat(3.14, 'f', 2, 64)  // float64 → string: "3.14"
    b2, _ := strconv.ParseBool("true")       // "true"/"false"/"1"/"0" → bool
    _ = n; _ = err; _ = s3; _ = f; _ = fs; _ = b2
    _ = greeting

    // Unicode checks
    unicode.IsLetter('A')                     // true
    unicode.IsDigit('5')                      // true
    unicode.IsSpace(' ')                      // true
    unicode.IsUpper('A')                      // true
    unicode.ToLower('A')                      // 'a'
}
