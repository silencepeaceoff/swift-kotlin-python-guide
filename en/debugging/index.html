// ──────────────────────────────────────────────
// log/slog — built-in structured logger (Go 1.21+)
// RECOMMENDED in 2026 for most projects
// ──────────────────────────────────────────────
func setupLogger() *slog.Logger {
    // JSON handler — for production (parsed by Grafana, Loki, Datadog)
    jsonHandler := slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
        Level:     slog.LevelDebug,            // minimum level
        AddSource: true,                       // add file:line to each log
    })

    // Text handler — for development (readable)
    textHandler := slog.NewTextHandler(os.Stderr, &slog.HandlerOptions{
        Level: slog.LevelDebug,
    })

    // Choose by environment
    var handler slog.Handler
    if os.Getenv("ENV") == "production" {
        handler = jsonHandler
    } else {
        handler = textHandler
    }

    logger := slog.New(handler)
    slog.SetDefault(logger)                    // set as global
    return logger
}

func loggingPatterns(log *slog.Logger) {
    // Basic levels: Debug, Info, Warn, Error
    log.Debug("debug message", "key", "value")
    log.Info("server started", "addr", ":8080", "env", "production")
    log.Warn("slow query", "duration", 500*time.Millisecond, "query", "SELECT ...")
    log.Error("database error", "err", err, "table", "users")

    // Structured attributes
    log.Info("user action",
        slog.String("user_id", "u-123"),
        slog.Int("attempt", 3),
        slog.Bool("success", true),
        slog.Duration("elapsed", 42*time.Millisecond),
        slog.Time("ts", time.Now()),
    )

    // Group — nested attributes
    log.Info("http request",
        slog.Group("request",
            slog.String("method", "GET"),
            slog.String("path", "/api/users"),
            slog.Int("status", 200),
        ),
        slog.Group("user",
            slog.String("id", "u-123"),
            slog.String("role", "admin"),
        ),
    )

    // With — preset attributes (for a service logger)
    serviceLog := log.With(
        slog.String("service", "user-service"),
        slog.String("version", "1.2.3"),
    )
    serviceLog.Info("processing request") // always includes service and version

    // Condition check — avoid computing expensive arguments
    if log.Enabled(context.Background(), slog.LevelDebug) {
        log.Debug("expensive debug", "data", expensiveSerialize())
    }
}

// ──────────────────────────────────────────────
// Context-aware logger — add request_id to each line
// ──────────────────────────────────────────────
type ctxKey string
const loggerKey ctxKey = "logger"

func WithLogger(ctx context.Context, log *slog.Logger) context.Context {
    return context.WithValue(ctx, loggerKey, log)
}

func LoggerFromCtx(ctx context.Context) *slog.Logger {
    if log, ok := ctx.Value(loggerKey).(*slog.Logger); ok {
        return log
    }
    return slog.Default()
}

// Middleware — add request_id to the logger in context
func requestLoggerMiddleware(log *slog.Logger, next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        requestID := r.Header.Get("X-Request-ID")
        if requestID == "" {
            requestID = generateID()
        }
        reqLog := log.With(
            slog.String("request_id", requestID),
            slog.String("method", r.Method),
            slog.String("path", r.URL.Path),
        )
        ctx := WithLogger(r.Context(), reqLog)
        reqLog.Info("request started")
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

// ──────────────────────────────────────────────
// Custom slog.Handler — for example, to send to Sentry
// ──────────────────────────────────────────────
type MultiHandler struct {
    handlers []slog.Handler
}

func NewMultiHandler(handlers ...slog.Handler) *MultiHandler {
    return &MultiHandler{handlers: handlers}
}

func (m *MultiHandler) Enabled(ctx context.Context, level slog.Level) bool {
    for _, h := range m.handlers {
        if h.Enabled(ctx, level) { return true }
    }
    return false
}

func (m *MultiHandler) Handle(ctx context.Context, r slog.Record) error {
    for _, h := range m.handlers {
        if h.Enabled(ctx, r.Level) {
            if err := h.Handle(ctx, r); err != nil { return err }
        }
    }
    return nil
}

func (m *MultiHandler) WithAttrs(attrs []slog.Attr) slog.Handler {
    handlers := make([]slog.Handler, len(m.handlers))
    for i, h := range m.handlers { handlers[i] = h.WithAttrs(attrs) }
    return NewMultiHandler(handlers...)
}

func (m *MultiHandler) WithGroup(name string) slog.Handler {
    handlers := make([]slog.Handler, len(m.handlers))
    for i, h := range m.handlers { handlers[i] = h.WithGroup(name) }
    return NewMultiHandler(handlers...)
}

// pprof — profiling (built into Go!)
// ──────────────────────────────────────────────
import _ "net/http/pprof"   // importing is enough!

// After adding the import:
// curl http://localhost:6060/debug/pprof/         — list of profiles
// go tool pprof http://localhost:6060/debug/pprof/heap   — heap profile
// go tool pprof http://localhost:6060/debug/pprof/goroutine  — goroutines

func startProfiler() {
    go func() {
        slog.Info("pprof listening", "addr", "localhost:6060")
        http.ListenAndServe("localhost:6060", nil)
    }()
}

// CPU profile through code
func profileCPU(fn func()) {
    f, _ := os.Create("cpu.prof")
    defer f.Close()
    pprof.StartCPUProfile(f)
    defer pprof.StopCPUProfile()
    fn()
}

// Memory profile
func dumpHeap() {
    f, _ := os.Create("heap.prof")
    defer f.Close()
    runtime.GC()                               // run GC before the snapshot
    pprof.WriteHeapProfile(f)
}

// ──────────────────────────────────────────────
// debug.ReadBuildInfo — build information
// ──────────────────────────────────────────────
func printBuildInfo() {
    info, ok := debug.ReadBuildInfo()
    if !ok { return }
    slog.Info("build info",
        "go_version", info.GoVersion,
        "path",       info.Path,
        "main_ver",   info.Main.Version,
    )
    for _, dep := range info.Deps {
        slog.Debug("dependency", "module", dep.Path, "version", dep.Version)
    }
}

// ──────────────────────────────────────────────
// race detector — run with the -race flag
// ──────────────────────────────────────────────
// go test -race ./...
// go run -race main.go
// go build -race -o app_race .

// ──────────────────────────────────────────────
// Panic trace — structured recover
// ──────────────────────────────────────────────
func panicMiddleware(log *slog.Logger, next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if err := recover(); err != nil {
                stack := debug.Stack()
                log.Error("panic recovered",
                    "error", err,
                    "stack", string(stack),
                    "path",  r.URL.Path,
                )
                http.Error(w, "Internal Server Error", http.StatusInternalServerError)
            }
        }()
        next.ServeHTTP(w, r)
    })
}
