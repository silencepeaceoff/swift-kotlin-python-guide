    Conditions🔗

    let x = 42

// Standard if-else
if x > 0 { print("positive") }
else if x == 0 { print("zero") }
else { print("negative") }

// if as an expression (Swift 5.9+)
let label = if x > 0 { "positive" } else { "non-positive" }

// guard — early exit (avoid the pyramid of doom)
func process(value: Int?) {
    guard let v = value, v > 0 else { return }
    // v is available here and until the end of the function
    print("Processing \(v)")
}

// guard let shorthand (Swift 5.7+)
func handle(value: Int?) {
    guard let value else { return }
    // value is available as non-optional
    print(value)
}

// switch — exhaustive (the compiler requires all cases)
switch x {
case 0:
    print("zero")
case 1...10:
    print("small")
case let n where n < 0:
    print("negative: \(n)")
case _ where x.isMultiple(of: 2):
    print("large even")
default:
    print("other")
}

// switch as an expression (Swift 5.9+)
let desc = switch x {
    case 0:      "zero"
    case 1...10: "small"
    default:     "large"
}

// Optional binding
let opt: Int? = 42
if let val = opt { print(val) }              // regular unwrap
if let opt { print(opt) }                    // shorthand (Swift 5.7+)
let val = opt ?? 0                           // nil coalescing
let forced = opt!                            // force unwrap — crashes if nil!

// Optional chaining
struct User { var address: Address? }
struct Address { var city: String }
let city = user.address?.city ?? "Unknown"

    val x = 42

// if — is an expression
val label = if (x > 0) "positive" else "non-positive"
val abs = if (x >= 0) x else -x

// when — powerful switch replacement (also an expression!)
when (x) {
    0          -> println("zero")
    in 1..10   -> println("small")
    !in 0..100 -> println("out of range")
    else       -> println("large: $x")
}

val description = when {              // without an argument — chain of conditions
    x < 0    -> "negative"
    x == 0   -> "zero"
    x < 100  -> "small"
    else     -> "large"
}

// Smart cast: is checks the type AND casts automatically
fun describe(obj: Any) = when (obj) {
    is Int    -> "Int: $obj"
    is String -> "String length: ${obj.length}"  // obj is already String here!
    is List<*>-> "List size: ${obj.size}"
    null      -> "null"
    else      -> "Unknown: ${obj::class.simpleName}"
}

// Elvis operator (equivalent of ??)
val name: String? = null
val display = name ?: "Anonymous"

// Safe call + Elvis
val length = name?.length ?: 0

// let (to run a block if not null)
name?.let { println("Name is: $it") }

// also, apply, run, with, let — scope functions
val result = listOf(1,2,3)
    .also { println("Original: $it") }      // returns the object itself
    .map { it * 2 }
    .filter { it > 2 }
    .let { it.sum() }                       // transforms into another type

    x = 42

# if-elif-else
if x > 0:
    print("positive")
elif x == 0:
    print("zero")
else:
    print("negative")

# Ternary operator
label = "positive" if x > 0 else "non-positive"

# match-case (Python 3.10+ — structural pattern matching)
match x:
    case 0:
        print("zero")
    case int() as n if n < 0:              # guard
        print(f"negative: {n}")
    case int() as n if 1 <= n <= 10:
        print("small")
    case _:                                # default
        print(f"other: {x}")

# match with destructuring of structures
point = (1, 0)
match point:
    case (0, 0): print("origin")
    case (x, 0): print(f"x-axis at {x}")
    case (0, y): print(f"y-axis at {y}")
    case (x, y): print(f"point ({x}, {y})")

# match with classes (dataclass)
from dataclasses import dataclass

@dataclass
class Command:
    action: str
    target: str

cmd = Command("open", "file.txt")
match cmd:
    case Command(action="open", target=t): print(f"Opening: {t}")
    case Command(action="delete"):         print("Deleting")
    case _:                                print("Unknown command")

# match with a list
match [1, 2, 3]:
    case []:          print("empty")
    case [x]:         print(f"one: {x}")
    case [x, y]:      print(f"two: {x}, {y}")
    case [x, *rest]:  print(f"first: {x}, rest: {rest}")

    package main

import (
    "fmt"
    "errors"
)

func controlFlow() {
    // ──────────────────────────────────────────
    // if / else if / else
    // ──────────────────────────────────────────
    x := 42

    // Init statement (the variable lives only inside the if)
    if y := x * 2; y > 100 {
        fmt.Println("large:", y)
    } else {
        fmt.Println("small:", y)
    }

    // ──────────────────────────────────────────
    // switch — no fallthrough by default
    // ──────────────────────────────────────────
    switch x {
    case 1, 2, 3:
        fmt.Println("small")
    case 42:
        fmt.Println("the answer")
    default:
        fmt.Println("other")
    }

    // switch without an expression = if-else chain
    switch {
    case x < 0:
        fmt.Println("negative")
    case x == 0:
        fmt.Println("zero")
    case x > 0:
        fmt.Println("positive")
    }

    // fallthrough — explicitly continue to the next case
    switch x {
    case 42:
        fmt.Println("42")
        fallthrough                            // execute the next case as well
    case 43:
        fmt.Println("43 or fallthrough from 42")
    }

    for i in 0..<5  { print(i) }                 // 0,1,2,3,4

for (index, value) in arr.enumerated() {

}
for value in arr where value > 15 { print(value) }   // with filter!

var n = 0
while n < 5 { n += 1 }
repeat { n -= 1 } while n > 0             // do-while equivalent

// labeled break (exit from a nested loop)
outer: for i in 0..<3 {
    for j in 0..<3 {
        if j == 1 { break outer }
    }
}

    for (i in 0 until 5)          { print(i) }  // 0,1,2,3,4
for (i in 0..5)               { print(i) }  // 0,1,2,3,4,5
for (i in 10 downTo 0 step 2) { print(i) }  // 10,8,6,4,2,0

val arr = listOf(10, 20, 30)
for ((index, value) in arr.withIndex()) { println("$index: $value") }
arr.forEach { println(it) }
arr.forEachIndexed { i, v -> println("$i: $v") }

var n = 0
while (n < 5) { n++ }
do { n-- } while (n > 0)

repeat(3) { println("hello $it") }          // built-in utility

// labeled break/continue
outer@ for (i in 0..3) {
    for (j in 0..3) {
        if (j == 1) break@outer
        if (j == 0) continue@outer
    }
}

# Iteration over multiple lists
for a, b in zip([1,2,3], ["a","b","c"]): print(a, b)
# zip_longest — if lengths differ
from itertools import zip_longest
for a, b in zip_longest([1,2], ["a","b","c"], fillvalue=0): print(a, b)

# for-else (unique Python feature)
for i in range(10):
    if i == 5: break
else:
    print("The loop finished without break")   # will NOT run (there was a break)

# Comprehension instead of a loop (faster and pythonic)
squares  = [x**2 for x in range(10)]
evens    = [x for x in range(20) if x % 2 == 0]
matrix   = [[i*j for j in range(5)] for i in range(5)]   # nested

# Generator expression — lazy version (does not load into memory)
total = sum(x**2 for x in range(1000000))   # does not create a list!

    // for — the only loop in Go (three forms)
    // ──────────────────────────────────────────

    // Classic C style
    for i := 0; i < 5; i++ {
        fmt.Println(i)
    }

    // while style
    n := 0
    for n < 10 {
        n += 2
    }

    // Infinite loop
    count := 0
    for {
        count++
        if count >= 3 { break }
    }

    // range — iteration over slice, map, channel, string, int (Go 1.22+)
    nums := []int{10, 20, 30}
    for i, v := range nums { fmt.Println(i, v) }
    for _, v := range nums { fmt.Println(v) }   // value only
    for i := range nums    { fmt.Println(i) }   // index only

    // range over integer (Go 1.22+)
    for i := range 5 {                          // 0, 1, 2, 3, 4
        fmt.Println(i)
    }

    // range over map
    m := map[string]int{"a": 1, "b": 2}
    for k, v := range m { fmt.Println(k, v) }

    // range over string → (byte index, rune)
    for i, r := range "Hello" { fmt.Println(i, r) }

    // range over channel
    ch := make(chan int, 3)
    ch <- 1; ch <- 2; ch <- 3; close(ch)
    for v := range ch { fmt.Println(v) }

    // break / continue with a label (for nested loops)
outer:
    for i := 0; i < 3; i++ {
        for j := 0; j < 3; j++ {
            if i == 1 && j == 1 {
                break outer                    // exit from the outer loop
            }
        }
    }

    // ──────────────────────────────────────────
    // defer — call after function return
    // ──────────────────────────────────────────
    func example() {
        defer fmt.Println("3: cleanup")       // runs last
        defer fmt.Println("2: after")         // LIFO order!
        fmt.Println("1: work")
    }
    // Output: 1: work → 2: after → 3: cleanup

    // defer + closure for changing named return
    func divide(a, b float64) (result float64, err error) {
        defer func() {
            if r := recover(); r != nil {
                err = fmt.Errorf("recovered: %v", r)
            }
        }()
        if b == 0 {
            return 0, errors.New("division by zero")
        }
        return a / b, nil
    }

    // panic / recover
    func safeDivide(a, b int) (result int, err error) {
        defer func() {
            if r := recover(); r != nil {
                err = fmt.Errorf("panic: %v", r)
            }
        }()
        return a / b, nil                      // panic when b==0
    }
