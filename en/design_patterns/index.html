    // Protocol-oriented DI (pure Swift)
protocol UserRepository: Sendable {
    func findById(_ id: Int) async throws -> User?
    func save(_ user: User) async throws
}

// Real implementation
final class SwiftDataUserRepository: UserRepository {
    private let container: ModelContainer

    init(container: ModelContainer) { self.container = container }

    func findById(_ id: Int) async throws -> User? {
        let context = container.mainContext
        let descriptor = FetchDescriptor<User>(
            predicate: #Predicate { $0.id == id }
        )
        return try context.fetch(descriptor).first
    }

    func save(_ user: User) async throws {
        let context = container.mainContext
        context.insert(user)
        try context.save()
    }
}

// Mock for tests
actor InMemoryUserRepository: UserRepository {
    private var users: [Int: User] = [:]

    func findById(_ id: Int) async throws -> User? { users[id] }
    func save(_ user: User) async throws { users[user.id] = user }
}

// Service depends on the protocol
@MainActor
class UserService: ObservableObject {
    private let repo: any UserRepository

    init(repo: any UserRepository) { self.repo = repo }

    func getUser(_ id: Int) async throws -> User {
        guard let user = try await repo.findById(id) else {
            throw AppError.notFound(id: "\(id)")
        }
        return user
    }
}

// DI Container pattern
@MainActor
final class AppContainer {
    lazy var modelContainer = try! ModelContainer(for: User.self)
    lazy var userRepo: any UserRepository = SwiftDataUserRepository(container: modelContainer)
    lazy var userService = UserService(repo: userRepo)
}

    # ── FastAPI Depends (built-in DI) ──
from fastapi import FastAPI, Depends
from sqlalchemy.ext.asyncio import AsyncSession

app = FastAPI()

async def get_session() -> AsyncSession:
    async with AsyncSessionLocal() as session:
        yield session                          # cleanup is automatic

async def get_user_repo(session: AsyncSession = Depends(get_session)):
    return SQLAlchemyUserRepository(session)

@app.get("/users/{user_id}")
async def get_user(
    user_id: int,
    repo: UserRepository = Depends(get_user_repo)
) -> User:
    user = await repo.find_by_id(user_id)
    if not user:
        raise HTTPException(status_code=404)
    return user

# ── Pure Python DI (without libraries) ──
from abc import ABC, abstractmethod
from dataclasses import dataclass

@dataclass
class User:
    id: int
    name: str
    email: str

class UserRepository(ABC):
    @abstractmethod
    async def find_by_id(self, uid: int) -> User | None: ...

class SQLAlchemyUserRepository(UserRepository):
    def __init__(self, session: AsyncSession) -> None:
        self._session = session

    async def find_by_id(self, uid: int) -> User | None:
        return await self._session.get(User, uid)

class InMemoryUserRepository(UserRepository):
    def __init__(self) -> None:
        self._users: dict[int, User] = {}

    async def find_by_id(self, uid: int) -> User | None:
        return self._users.get(uid)

class UserService:
    def __init__(self, repo: UserRepository) -> None:
        self._repo = repo

    async def get_user(self, uid: int) -> User:
        user = await self._repo.find_by_id(uid)
        if user is None:
            raise ValueError(f"User {uid} not found")
        return user

# Production
service = UserService(repo=SQLAlchemyUserRepository(session))

# Test
async def test_get_user() -> None:
    repo = InMemoryUserRepository()
    repo._users[1] = User(id=1, name="Alice", email="alice@example.com")
    service = UserService(repo=repo)
    user = await service.get_user(1)
    assert user.name == "Alice"

// ① Singleton — via sync.Once (thread-safe)
// ──────────────────────────────────────────────
type DB struct{ pool *sql.DB }

var (
    dbInstance *DB
    dbOnce     sync.Once
)

func GetDB() *DB {
    dbOnce.Do(func() {
        pool, err := sql.Open("postgres", os.Getenv("DATABASE_URL"))
        if err != nil { panic(err) }
        dbInstance = &DB{pool: pool}
    })
    return dbInstance
}

// ──────────────────────────────────────────────
// ② Observer — through channels and callbacks
// ──────────────────────────────────────────────
type EventBus[T any] struct {
    mu          sync.RWMutex
    subscribers map[string][]chan T
}

func NewEventBus[T any]() *EventBus[T] {
    return &EventBus[T]{subscribers: make(map[string][]chan T)}
}

func (eb *EventBus[T]) Subscribe(topic string, bufSize int) (<-chan T, func()) {
    ch := make(chan T, bufSize)
    eb.mu.Lock()
    eb.subscribers[topic] = append(eb.subscribers[topic], ch)
    idx := len(eb.subscribers[topic]) - 1
    eb.mu.Unlock()

    unsubscribe := func() {
        eb.mu.Lock()
        defer eb.mu.Unlock()
        subs := eb.subscribers[topic]
        subs[idx] = subs[len(subs)-1]
        eb.subscribers[topic] = subs[:len(subs)-1]
        close(ch)
    }
    return ch, unsubscribe
}

func (eb *EventBus[T]) Publish(ctx context.Context, topic string, event T) {
    eb.mu.RLock()
    subs := eb.subscribers[topic]
    eb.mu.RUnlock()

    for _, ch := range subs {
        select {
        case ch <- event:
        case <-ctx.Done():
            return
        default:                              // do not block if the buffer is full
            slog.Warn("subscriber buffer full", "topic", topic)
        }
    }
}

// Usage
bus := NewEventBus[UserEvent]()
ch, unsub := bus.Subscribe("user.login", 10)
defer unsub()

go func() {
    for event := range ch {
        slog.Info("user logged in", "user_id", event.UserID)
    }
}()

bus.Publish(ctx, "user.login", UserEvent{UserID: "u-123"})

// ──────────────────────────────────────────────
// ③ Factory — through a map of functions
// ──────────────────────────────────────────────
type Serializer interface {
    Marshal(v any) ([]byte, error)
    Unmarshal(data []byte, v any) error
}

type JSONSerializer  struct{}
type XMLSerializer   struct{}
type YAMLSerializer  struct{}

func (s *JSONSerializer) Marshal(v any) ([]byte, error)          { return json.Marshal(v) }
func (s *JSONSerializer) Unmarshal(data []byte, v any) error     { return json.Unmarshal(data, v) }
func (s *XMLSerializer) Marshal(v any) ([]byte, error)           { return xml.Marshal(v) }
func (s *XMLSerializer) Unmarshal(data []byte, v any) error      { return xml.Unmarshal(data, v) }

type SerializerFactory struct {
    registry map[string]func() Serializer
}

func NewSerializerFactory() *SerializerFactory {
    f := &SerializerFactory{registry: make(map[string]func() Serializer)}
    f.Register("json", func() Serializer { return &JSONSerializer{} })
    f.Register("xml",  func() Serializer { return &XMLSerializer{} })
    f.Register("yaml", func() Serializer { return &YAMLSerializer{} })
    return f
}

func (f *SerializerFactory) Register(name string, ctor func() Serializer) {
    f.registry[name] = ctor
}

func (f *SerializerFactory) Create(name string) (Serializer, error) {
    ctor, ok := f.registry[name]
    if !ok { return nil, fmt.Errorf("unknown serializer: %q", name) }
    return ctor(), nil
}

factory := NewSerializerFactory()
s, _ := factory.Create("json")
data, _ := s.Marshal(user)

// ──────────────────────────────────────────────
// ④ Decorator — via interface wrapping
// ──────────────────────────────────────────────
type UserService interface {
    GetUser(ctx context.Context, id int) (*User, error)
    CreateUser(ctx context.Context, name, email string) (*User, error)
}

// Decorator: metrics
type MetricsService struct {
    inner   UserService
    metrics *Metrics
}

func (s *MetricsService) GetUser(ctx context.Context, id int) (*User, error) {
    start := time.Now()
    defer func() {
        s.metrics.RecordDuration("get_user", time.Since(start))
    }()
    return s.inner.GetUser(ctx, id)
}

// Decorator: caching
type CachingService struct {
    inner UserService
    cache *Cache[int, *User]
}

func (s *CachingService) GetUser(ctx context.Context, id int) (*User, error) {
    if user, ok := s.cache.Get(id); ok { return user, nil }
    user, err := s.inner.GetUser(ctx, id)
    if err == nil { s.cache.Set(id, user) }
    return user, err
}

// Composing
svc := &MetricsService{
    inner: &CachingService{
        inner: &RealUserService{repo: userRepo},
        cache: NewCache[int, *User](5*time.Minute),
    },
    metrics: metrics,
}

// ⑤ DI — through Constructor Injection
// ──────────────────────────────────────────────
// Go has no default DI framework.
// Wire (google/wire) — compile-time DI (go get github.com/google/wire)
// Uber/dig — runtime DI (go get go.uber.org/dig)

// Pure Constructor Injection (recommended for most projects)
type App struct {
    server  *http.Server
    db      *sql.DB
    logger  *slog.Logger
}

func NewApp(cfg *Config) (*App, error) {
    logger := setupLogger(cfg.Debug)

    db, err := setupDB(cfg.DatabaseURL)
    if err != nil { return nil, fmt.Errorf("setup db: %w", err) }

    userRepo := NewUserRepository(db)
    userSvc  := NewUserService(userRepo)
    handler  := NewHandler(userSvc, logger)
    router   := NewRouter(handler, logger)

    server := &http.Server{
        Addr:    fmt.Sprintf(":%d", cfg.Port),
        Handler: router,
    }

    return &App{server: server, db: db, logger: logger}, nil
}

func (a *App) Run(ctx context.Context) error {
    a.logger.Info("starting", "addr", a.server.Addr)
    go func() {
        <-ctx.Done()
        a.server.Shutdown(context.Background())
    }()
    if err := a.server.ListenAndServe(); !errors.Is(err, http.ErrServerClosed) {
        return err
    }
    return nil
}
