}

// ──────────────────────────────────────────────
// Базовые операции — os пакет
// ──────────────────────────────────────────────
func fileBasics() error {
    // ─── Запись файла ─────────────────────────
    // os.WriteFile — атомарная запись (через temp файл на большинстве FS)
    data := []byte("Hello, Go!\n")
    if err := os.WriteFile("output.txt", data, 0644); err != nil {
        return fmt.Errorf("write file: %w", err)
    }

    // Запись через os.Create + bufio.Writer (эффективно для больших файлов)
    f, err := os.Create("large.txt")
    if err != nil { return err }
    defer f.Close()

    bw := bufio.NewWriterSize(f, 64*1024)      // 64 KB буфер
    for i := range 100_000 {
        fmt.Fprintf(bw, "line %d\n", i)
    }
    if err := bw.Flush(); err != nil { return err } // сбросить буфер!

    // ─── Чтение файла ─────────────────────────
    content, err := os.ReadFile("output.txt") // весь файл в память
    if err != nil { return err }
    fmt.Println(string(content))

    // Побайтовое/построчное чтение (большие файлы)
    f2, err := os.Open("large.txt")
    if err != nil { return err }
    defer f2.Close()

    scanner := bufio.NewScanner(f2)
    scanner.Buffer(make([]byte, 1<<20), 1<<20) // буфер 1 MB для длинных строк
    lineCount := 0
    for scanner.Scan() {
        line := scanner.Text()
        _ = line
        lineCount++
    }
    if err := scanner.Err(); err != nil { return err }
    fmt.Println("lines:", lineCount)

    // ─── Добавление в конец файла ─────────────
    f3, err := os.OpenFile("log.txt",
        os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0644)
    if err != nil { return err }
    defer f3.Close()
    fmt.Fprintln(f3, time.Now().Format(time.RFC3339), "app started")

    // ─── Проверка существования / stat ────────
    info, err := os.Stat("output.txt")
    if err != nil {
        if errors.Is(err, os.ErrNotExist) {
            fmt.Println("file not found")
        }
        return err
    }
    fmt.Printf("size=%d mode=%s modtime=%s\n",
        info.Size(), info.Mode(), info.ModTime())

    // ─── Переименование и удаление ────────────
    os.Rename("output.txt", "output_new.txt")
    os.Remove("output_new.txt")
    os.RemoveAll("temp_dir")               // рекурсивно

    return nil
}

// Работа с директориями

func dirOperations() error {
    // Создать директории
    os.MkdirAll("data/cache/tmp", 0755)    // как mkdir -p

    // Перечислить директорию
    entries, err := os.ReadDir(".")
    if err != nil { return err }
    for _, e := range entries {
        info, _ := e.Info()
        fmt.Printf("%-30s %8d %s\n", e.Name(), info.Size(), info.Mode())
    }

    // Рекурсивный обход — filepath.WalkDir (Go 1.16+, эффективнее Walk)
    err = filepath.WalkDir(".", func(path string, d fs.DirEntry, err error) error {
        if err != nil { return err }

        if d.IsDir() && d.Name() == ".git" {
            return filepath.SkipDir  // пропустить поддерево
        }

        if !d.IsDir() && strings.HasSuffix(path, ".go") {
            info, _ := d.Info()
            fmt.Printf("%s (%d bytes)\n", path, info.Size())
        }
        return nil
    })

    // Temp файлы — автоматически безопасное имя
    tmpFile, err := os.CreateTemp("", "app-*.tmp")  // * заменяется случайной строкой
    if err != nil { return err }
    defer os.Remove(tmpFile.Name())                  // удалить при завершении
    defer tmpFile.Close()
    tmpFile.WriteString("temp data")

    // Temp директория
    tmpDir, err := os.MkdirTemp("", "app-*")
    if err != nil { return err }
    defer os.RemoveAll(tmpDir)

    _ = err
    return nil
}

// ──────────────────────────────────────────────
// Атомарная запись файла — через rename
// (защита от потери данных при краше в середине записи)
// ──────────────────────────────────────────────
func writeFileAtomic(path string, data []byte, perm os.FileMode) error {
    dir := filepath.Dir(path)

    // Создать временный файл в той же ФС (важно для atomic rename!)
    tmp, err := os.CreateTemp(dir, ".tmp-*")
    if err != nil { return fmt.Errorf("create temp: %w", err) }
    tmpName := tmp.Name()

    defer func() {
        tmp.Close()
        os.Remove(tmpName)                   // cleanup если что-то пошло не так
    }()

    if _, err := tmp.Write(data); err != nil {
        return fmt.Errorf("write temp: %w", err)
    }
    if err := tmp.Sync(); err != nil {       // fsync — сбросить на диск!
        return fmt.Errorf("sync: %w", err)
    }
    tmp.Close()

    if err := os.Chmod(tmpName, perm); err != nil { return err }

    // Atomic rename: либо старый, либо новый — никогда частичный
    return os.Rename(tmpName, path)
}

// CSV — encoding/csv (встроен)

type Record struct {
    Name  string
    Email string
    Age   int
}

func writeCSV(path string, records []Record) error {
    f, err := os.Create(path)
    if err != nil { return err }
    defer f.Close()

    w := csv.NewWriter(f)
    defer w.Flush()                          // ОБЯЗАТЕЛЬНО Flush!

    // Заголовок
    if err := w.Write([]string{"name", "email", "age"}); err != nil {
        return err
    }

    for _, r := range records {
        if err := w.Write([]string{
            r.Name, r.Email, strconv.Itoa(r.Age),
        }); err != nil { return err }
    }
    return w.Error()
}

func readCSV(path string) ([]Record, error) {
    f, err := os.Open(path)
    if err != nil { return nil, err }
    defer f.Close()

    r := csv.NewReader(f)
    r.TrimLeadingSpace = true
    r.Comment = '#'                          // игнорировать строки с #

    // Пропустить заголовок
    if _, err := r.Read(); err != nil { return nil, err }

    var records []Record
    for {
        row, err := r.Read()
        if err == io.EOF { break }
        if err != nil { return nil, err }

        age, _ := strconv.Atoi(row[2])
        records = append(records, Record{
            Name:  row[0],
            Email: row[1],
            Age:   age,
        })
    }
    return records, nil
}

// ──────────────────────────────────────────────
// JSON streaming — для больших файлов
// ──────────────────────────────────────────────
func streamLargeJSON(path string, process func(User) error) error {
    f, err := os.Open(path)
    if err != nil { return err }
    defer f.Close()

    dec := json.NewDecoder(bufio.NewReaderSize(f, 64*1024))

    // Пропустить '[' открывающую скобку
    if _, err := dec.Token(); err != nil { return err }

    processed := 0
    for dec.More() {                         // More() — есть ещё элементы?
        var user User
        if err := dec.Decode(&user); err != nil {
            return fmt.Errorf("decode item %d: %w", processed, err)
        }
        if err := process(user); err != nil { return err }
        processed++
    }
    fmt.Printf("processed %d records\n", processed)
    return nil
}
