    Speicherung im Speicher🔗

    Konzept Swift Kotlin Python
    Value Types struct, enum, Basistypen (Stack / Inline auf Heap) Int, Boolean usw. kompilieren als JVM-Primitive — (alles ist ein Objekt)
    Reference Types class (ARC auf dem Heap) class, alle Collections (JVM Heap) Alle Objekte (int, str, list — auf dem Heap)
    Speicherverwaltung ARC (Automatic Reference Counting) — deterministisch JVM GC (G1 / ZGC / Shenandoah) CPython: Ref-Counting + zyklischer GC (gc Modul)
    Null-Safety Kompilierzeit (Optional) Kompilierzeit (?) Laufzeit (None)
    Kopierverhalten Value Types kopieren bei Zuweisung (COW-Optimierung) Objekte werden als Referenz übergeben Alle Objekte per Referenz; Zahlen/Strings — unveränderlich

    CPython Integer Caching

    CPython speichert int Objekte im Bereich von −5 … 256 zwischen und internt kurze Bezeichner. Das bedeutet, a = 256; b = 256; a is b gibt True zurück (selbes Objekt), während a = 257; b = 257; a is b False liefert (zwei verschiedene Objekte). Verwenden Sie immer == statt is, um Werte zu vergleichen.

    package main

import (
    "fmt"
    "math"
    "math/big"
    "math/bits"
)

func numbers() {
    // ──────────────────────────────────────────
    // Ganzzahlen
    // ──────────────────────────────────────────
    var a int   = 100
    var b int64 = 9_223_372_036_854_775_807  // Trennzeichen _ zur besseren Lesbarkeit (Go 1.13+)
    var c uint  = 0xFF                        // Hex-Literal
    var d int   = 0b1010_1010                 // Binär-Literal (Go 1.13+)
    var e int   = 0o755                       // Oktal-Literal (Go 1.13+)
    _ = a; _ = b; _ = c; _ = d; _ = e

    // Ganzzahlarithmetik
    fmt.Println(10 / 3)                       // 3 (abgeschnitten)
    fmt.Println(10 % 3)                       // 1
    fmt.Println(-7 % 3)                       // -1 (Vorzeichen entspricht dem Dividenden)

    // ──────────────────────────────────────────
    // Gleitkommazahlen
    // ──────────────────────────────────────────
    x := 3.14
    y := 2.71828
    fmt.Printf("%.4f\n", x + y)              // Formatierung

    // Spezielle Werte
    posInf := math.Inf(1)                     // +∞
    negInf := math.Inf(-1)                    // -∞
    nan    := math.NaN()
    fmt.Println(math.IsInf(posInf, 1))        // true
    fmt.Println(math.IsNaN(nan))              // true

    // Float-Vergleich — NUR über Epsilon!
    const eps = 1e-9
    func floatEqual(a, b float64) bool { return math.Abs(a-b) < eps }

    // ──────────────────────────────────────────
    // Mathematische Funktionen
    // ──────────────────────────────────────────
    math.Abs(-5.0)                            // 5.0
    math.Sqrt(16.0)                           // 4.0
    math.Pow(2, 10)                           // 1024.0
    math.Log(math.E)                          // 1.0
    math.Log2(1024)                           // 10.0
    math.Log10(1000)                          // 3.0
    math.Floor(3.9)                           // 3.0
    math.Ceil(3.1)                            // 4.0
    math.Round(3.5)                           // 4.0 (Banker-Rundung: Round(2.5) = 2)
    math.Max(3.0, 5.0)                        // 5.0
    math.Min(3.0, 5.0)                        // 3.0

    // Trigonometrie
    math.Sin(math.Pi / 2)                     // 1.0
    math.Cos(0)                               // 1.0

    // ──────────────────────────────────────────
    // Bit-Operationen
    // ──────────────────────────────────────────
    n := 0b1010
    fmt.Println(n & 0b0110)                   // AND: 0b0010 = 2
    fmt.Println(n | 0b0110)                   // OR:  0b1110 = 14
    fmt.Println(n ^ 0b0110)                   // XOR: 0b1100 = 12
    fmt.Println(n << 2)                       // Left Shift: 40
    fmt.Println(n >> 1)                       // Right Shift: 5
    fmt.Println(^n)                           // NOT: -11 (Bitweises Komplement)
    fmt.Println(n &^ 0b0110)                  // AND NOT (Bit löschen): 0b1000 = 8

    // math/bits — Bit-Zählung (Go 1.9+)
    bits.OnesCount(0b1010_1010)               // 4 (gesetzte Bits)
    bits.Len(255)                             // 8 (Bit-Länge)
    bits.LeadingZeros(1)                      // 63 (für uint64)

    // ──────────────────────────────────────────
    // big.Int — beliebige Genauigkeit
    // ──────────────────────────────────────────
    a_big := new(big.Int).SetInt64(1_000_000_000_000)
    b_big := new(big.Int).SetInt64(999_999_999_999)
    sum   := new(big.Int).Add(a_big, b_big)
    fmt.Println(sum.String())                 // "1999999999999"

    // Fakultät 100
    result := new(big.Int).SetInt64(1)
    for i := int64(2); i <= 100; i++ {
        result.Mul(result, big.NewInt(i))
    }
    fmt.Println(result.String())              // 93326215443944152681699238856266...

    // big.Float — Gleitkomma mit beliebiger Genauigkeit
    pi, _, _ := big.ParseFloat("3.14159265358979323846", 10, 256, big.ToNearestEven)
    fmt.Println(pi.Text('f', 30))             // 30 Nachkommastellen
}

func stringBasics() {
    // ──────────────────────────────────────────
    // Literale
    // ──────────────────────────────────────────
    s  := "Hello, 世界"           // Standardmäßig UTF-8
    raw := `Multiline
raw string \n wird nicht maskiert`// Raw String Literal (Backticks)

    // Länge: Bytes vs Runen!
    fmt.Println(len(s))                       // 13 Bytes (NICHT Zeichen!)
    fmt.Println(utf8.RuneCountInString(s))    // 9 Runen (Zeichen)

    // Iteration — IMMER range für Unicode verwenden
    for i, r := range s {
        fmt.Printf("pos=%d rune=%c (%d)\n", i, r, r)
    }

    // Explizite Konvertierung string ↔ []rune (für Manipulationen)
    runes := []rune(s)
    runes[0] = 'h'
    modified := string(runes)                 // "hello, 世界"

    // string ↔ []byte (für binäre Operationen)
    bytes := []byte(s)
    bytes[0] = 'H'
    _ = string(bytes)                         // "Hello, 世界"
    _ = raw; _ = modified

    // ──────────────────────────────────────────
    // Formatierung (fmt.Sprintf)
    // ──────────────────────────────────────────
    name, age, score := "Alice", 30, 98.5
    fmt.Sprintf("Name: %s, Age: %d, Score: %.2f", name, age, score)
    fmt.Sprintf("Binary: %b, Hex: %x, Oct: %o", 42, 255, 8)
    fmt.Sprintf("Width: %10s, %-10s", "right", "left")   // Ausrichtung
    fmt.Sprintf("Padded: %05d", 42)           // "00042"
    fmt.Sprintf("Pointer: %p", &name)
    fmt.Sprintf("Type: %T, Value: %v", name, name)  // "string", "Alice"
    fmt.Sprintf("Struct: %+v", struct{ X, Y int }{1, 2})  // {X:1 Y:2}

    // ──────────────────────────────────────────
    // strings Package — Basisoperationen
    // ──────────────────────────────────────────
    s2 := "  Hello, World!  "

    strings.TrimSpace(s2)                     // "Hello, World!"
    strings.Trim(s2, " !")                    // "Hello, World"
    strings.TrimLeft(s2, " ")                 // "Hello, World!  "
    strings.TrimRight(s2, " ")                // "  Hello, World!"
    strings.TrimPrefix("photo.jpg", "photo.") // "jpg"
    strings.TrimSuffix("photo.jpg", ".jpg")   // "photo"

    strings.ToUpper("hello")                  // "HELLO"
    strings.ToLower("HELLO")                  // "hello"
    // strings.Title ist veraltet, stattdessen cases.Title verwenden

    strings.HasPrefix("filename.go", "file")  // true
    strings.HasSuffix("filename.go", ".go")   // true
    strings.Contains("hello world", "world")  // true
    strings.ContainsAny("hello", "aeiou")     // true (enthält mindestens einen der Buchstaben)
    strings.Count("cheese", "e")             // 3
    strings.Index("hello", "ll")             // 2 (-1 wenn nicht gefunden)
    strings.LastIndex("hello", "l")          // 3

    strings.Replace("oink oink oink", "oink", "moo", 2) // "moo moo oink"
    strings.ReplaceAll("foo bar foo", "foo", "baz")      // "baz bar baz"

    strings.Split("a,b,c", ",")              // ["a", "b", "c"]
    strings.SplitN("a,b,c", ",", 2)         // ["a", "b,c"]
    strings.Fields("  foo  bar  ")           // ["foo", "bar"] (am Leerzeichen aufteilen)
    strings.Join([]string{"a", "b", "c"}, "-") // "a-b-c"

    strings.Repeat("Go!", 3)                 // "Go!Go!Go!"

    // ──────────────────────────────────────────
    // strings.Builder — effiziente Konkatenation
    // ──────────────────────────────────────────
    var b strings.Builder
    for i := 0; i < 5; i++ {
        fmt.Fprintf(&b, "item%d ", i)
    }
    result := b.String()                      // "item0 item1 item2 item3 item4 "
    b.Reset()                                 // für erneute Verwendung leeren
    _ = result

    greeting := "Hello" + ", " + "World!"    // erstellt 2 Zwischenstrings

    // strconv — Konvertierungen String ↔ Zahlen
    // ──────────────────────────────────────────

    s3 := strconv.Itoa(42)                   // int → string
    f, _ := strconv.ParseFloat("3.14", 64)   // string → float64
    fs := strconv.FormatFloat(3.14, 'f', 2, 64)  // float64 → string: "3.14"
    b2, _ := strconv.ParseBool("true")       // "true"/"false"/"1"/"0" → bool
    _ = n; _ = err; _ = s3; _ = f; _ = fs; _ = b2
    _ = greeting

    // Unicode Prüfungen
    unicode.IsLetter('A')                     // true
    unicode.IsDigit('5')                      // true
    unicode.IsSpace(' ')                      // true
    unicode.IsUpper('A')                      // true
    unicode.ToLower('A')                      // 'a'
}
