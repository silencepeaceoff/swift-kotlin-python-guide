    # Pydantic v2 (de-facto-Standard 2026)
from pydantic import (BaseModel, Field, field_validator,
                      model_validator, computed_field, ConfigDict)
from datetime import datetime

class User(BaseModel):
    id: int
    name: str
    email: str
    created_at: datetime = Field(default_factory=datetime.utcnow)

    model_config = ConfigDict(
        frozen=False,
        str_strip_whitespace=True,
        validate_assignment=True,
        populate_by_name=True,
    )

class Article(BaseModel):
    id: int
    title: str = Field(min_length=3, max_length=200)
    author_name: str = Field(alias="authorName")
    tags: list[str] = []
    views: int = Field(default=0, ge=0)

    @field_validator("title")
    @classmethod
    def title_must_not_be_empty(cls, v: str) -> str:
        if not v.strip():
            raise ValueError("Title must not be empty")
        return v.strip()

    @computed_field
    @property
    def title_upper(self) -> str:
        return self.title.upper()

type User struct {
    ID        int       `json:"id"`
    Name      string    `json:"name"`
    Email     string    `json:"email"`
    Password  string    `json:"-"`               // niemals serialisieren!
    CreatedAt time.Time `json:"created_at"`
    UpdatedAt *time.Time `json:"updated_at,omitempty"` // omit wenn nil
    Tags      []string  `json:"tags,omitempty"`        // omit wenn leer

    // Go 1.24: omitzero — omit bei Zero Value des Typs
    Score int `json:"score,omitzero"`
}

// Eigene Serialisierung
type DateOnly struct{ time.Time }

func (d DateOnly) MarshalJSON() ([]byte, error) {
    return json.Marshal(d.Format("2006-01-02"))
}

func (d *DateOnly) UnmarshalJSON(data []byte) error {
    var s string
    if err := json.Unmarshal(data, &s); err != nil { return err }
    t, err := time.Parse("2006-01-02", s)
    if err != nil { return err }
    d.Time = t
    return nil
}

func jsonExamples() {
    user := User{ID: 1, Name: "Alice", Email: "alice@example.com",
        Tags: []string{"admin", "user"}}

    // Encode → []byte
    data, err := json.Marshal(user)
    // Encode mit Einrückung
    pretty, err := json.MarshalIndent(user, "", "  ")
    _ = err

    // Decode ← []byte
    var decoded User
    if err := json.Unmarshal(data, &decoded); err != nil {
        fmt.Println("decode error:", err)
    }

    // Streaming — für große Daten
    enc := json.NewEncoder(os.Stdout)
    enc.SetIndent("", "  ")
    enc.SetEscapeHTML(false)                   // <, > und & nicht escapen
    enc.Encode(user)

    dec := json.NewDecoder(os.Stdin)
    var v User
    dec.Decode(&v)

    // json.RawMessage — Dekodierung aufschieben
    type Response struct {
        Status string          `json:"status"`
        Data   json.RawMessage `json:"data"`   // ← später dekodieren
    }
    var resp Response
    json.Unmarshal([]byte(`{"status":"ok","data":{"id":1}}`), &resp)

    var user2 User
    json.Unmarshal(resp.Data, &user2)

    // Dynamisches JSON über map
    var m map[string]any
    json.Unmarshal(data, &m)
    name := m["name"].(string)
    _ = name; _ = pretty; _ = pretty
}

type Product struct {
    XMLName     xml.Name  `xml:"product"`
    ID          int       `xml:"id,attr"`     // Attribut
    Name        string    `xml:"name"`
    Price       float64   `xml:"price"`
    Description string    `xml:"description,omitempty"`
    Tags        []string  `xml:"tags>tag"`    // verschachtelter Tag
}

func xmlExamples() {
    p := Product{
        ID: 1, Name: "Go Book",
        Price: 29.99,
        Tags: []string{"go", "programming"},
    }

    // Encode
    data, _ := xml.MarshalIndent(p, "", "  ")
    fmt.Println(`<?xml version="1.0" encoding="UTF-8"?>`)
    fmt.Println(string(data))

    // Decode
    xmlStr := `<product id="2"><name>Rust Book</name><price>34.99</price></product>`
    var p2 Product
    xml.Unmarshal([]byte(xmlStr), &p2)

    // Streaming XML — für große Dokumente
    enc := xml.NewEncoder(os.Stdout)
    enc.Indent("", "  ")
    enc.Encode(p)
    enc.Flush()
}

type AppConfig struct {

}

func yamlExamples() {
    yamlStr := `
server:
  host: localhost
  port: 8080
database:
  url: postgres://localhost/mydb
  pool_size: 10
features:
  - auth
  - notifications
debug: false
`

    var cfg AppConfig
    if err := yaml.Unmarshal([]byte(yamlStr), &cfg); err != nil {
        panic(err)
    }
    fmt.Printf("host: %s, port: %d\n", cfg.Server.Host, cfg.Server.Port)

    // Marshal → YAML
    data, _ := yaml.Marshal(cfg)
    fmt.Println(string(data))

    // Aus Datei
    f, _ := os.Open("config.yaml")
    defer f.Close()
    yaml.NewDecoder(f).Decode(&cfg)

    // Dynamisches YAML (ohne Struct)
    var raw map[string]any
    yaml.Unmarshal([]byte(yamlStr), &raw)
}

type TomlConfig struct {
    Host  string   `toml:"host"`
    Port  int      `toml:"port"`
    Tags  []string `toml:"tags"`
    DB    struct {
        URL string `toml:"url"`
    } `toml:"database"`
}

func tomlExamples() {
    tomlStr := `
host = "localhost"
port = 8080
tags = ["production", "v2"]

[database]
url = "postgres://localhost/mydb"
`
    var cfg TomlConfig
    if _, err := toml.Decode(tomlStr, &cfg); err != nil { panic(err) }

    // Aus Datei
    if _, err := toml.DecodeFile("config.toml", &cfg); err != nil { panic(err) }

    // Encode → TOML
    f, _ := os.Create("output.toml")
    defer f.Close()
    toml.NewEncoder(f).Encode(cfg)
}
