// Typsicheres JSON-Enum für beliebige Strukturen
indirect enum JSON: Codable {
    case null
    case bool(Bool)
    case int(Int)
    case double(Double)
    case string(String)
    case array([JSON])
    case object([String: JSON])

    subscript(key: String) -> JSON? {
        if case .object(let dict) = self { return dict[key] }
        return nil
    }
    subscript(index: Int) -> JSON? {
        if case .array(let arr) = self, arr.indices.contains(index) {
            return arr[index]
        }
        return nil
    }
}

    import json

# Benutzerdefinierter Encoder für nicht standardmäßige Typen
class AppEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        if hasattr(obj, "__dict__"):
            return obj.__dict__
        return super().default(obj)

// JSON — encoding/json (eingebaut)

type User struct {
    ID        int       `json:"id"`
    Name      string    `json:"name"`
    Email     string    `json:"email"`
    Password  string    `json:"-"`               // niemals serialisieren!
    CreatedAt time.Time `json:"created_at"`
    UpdatedAt *time.Time `json:"updated_at,omitempty"` // weglassen, wenn nil
    Tags      []string  `json:"tags,omitempty"`        // weglassen, wenn leer

    // Go 1.24: omitzero — weglassen, wenn Zero Value des Typs
    Score int `json:"score,omitzero"`
}

// Benutzerdefinierte Serialisierung
type DateOnly struct{ time.Time }

func (d DateOnly) MarshalJSON() ([]byte, error) {
    return json.Marshal(d.Format("2006-01-02"))
}

func (d *DateOnly) UnmarshalJSON(data []byte) error {
    var s string
    if err := json.Unmarshal(data, &s); err != nil { return err }
    t, err := time.Parse("2006-01-02", s)
    if err != nil { return err }
    d.Time = t
    return nil
}

func jsonExamples() {
    user := User{ID: 1, Name: "Alice", Email: "alice@example.com",
        Tags: []string{"admin", "user"}}

    // Encode → []byte
    data, err := json.Marshal(user)
    // Encode mit Einrückung
    pretty, err := json.MarshalIndent(user, "", "  ")
    _ = err

    // Decode ← []byte
    var decoded User
    if err := json.Unmarshal(data, &decoded); err != nil {
        fmt.Println("decode error:", err)
    }

    // Streaming — für große Datenmengen
    enc := json.NewEncoder(os.Stdout)
    enc.SetIndent("", "  ")
    enc.SetEscapeHTML(false)                   // <, >, & nicht escapen
    enc.Encode(user)

    dec := json.NewDecoder(os.Stdin)
    var v User
    dec.Decode(&v)

    // json.RawMessage — Decoding verschieben
    type Response struct {
        Status string          `json:"status"`
        Data   json.RawMessage `json:"data"`   // ← später decodieren
    }
    var resp Response
    json.Unmarshal([]byte(`{"status":"ok","data":{"id":1}}`), &resp)

    var user2 User
    json.Unmarshal(resp.Data, &user2)

    // Dynamisches JSON über map
    var m map[string]any
    json.Unmarshal(data, &m)
    name := m["name"].(string)
    _ = name; _ = pretty; _ = pretty
}

    import Foundation

// XMLParser — SAX-Parser (eventbasiert, eingebaut)
class SimpleXMLParser: NSObject, XMLParserDelegate {
    var result: [String: String] = [:]
    private var currentElement = ""
    private var currentValue   = ""

    func parser(_ parser: XMLParser,
                didStartElement element: String,
                namespaceURI: String?,
                qualifiedName: String?,
                attributes: [String: String]) {
        currentElement = element
        currentValue   = ""
    }

    func parser(_ parser: XMLParser, foundCharacters string: String) {
        currentValue += string
    }

    func parser(_ parser: XMLParser,
                didEndElement element: String,
                namespaceURI: String?,
                qualifiedName: String?) {
        result[element] = currentValue.trimmingCharacters(in: .whitespaces)
    }
}

    // XML — encoding/xml (eingebaut)
// ──────────────────────────────────────────────
type Product struct {
    XMLName     xml.Name  `xml:"product"`
    ID          int       `xml:"id,attr"`     // Attribut
    Name        string    `xml:"name"`
    Price       float64   `xml:"price"`
    Description string    `xml:"description,omitempty"`
    Tags        []string  `xml:"tags>tag"`    // verschachtelter Tag
}

func xmlExamples() {
    p := Product{
        ID: 1, Name: "Go Book",
        Price: 29.99,
        Tags: []string{"go", "programming"},
    }

    // Encode
    data, _ := xml.MarshalIndent(p, "", "  ")
    fmt.Println(`<?xml version="1.0" encoding="UTF-8"?>`)
    fmt.Println(string(data))

    // Decode
    xmlStr := `<product id="2"><name>Rust Book</name><price>34.99</price></product>`
    var p2 Product
    xml.Unmarshal([]byte(xmlStr), &p2)

    // Streaming-XML — für große Dokumente
    enc := xml.NewEncoder(os.Stdout)
    enc.Indent("", "  ")
    enc.Encode(p)
    enc.Flush()
}
