// Operatorüberladung
data class Vec(val x: Int, val y: Int) {
    operator fun plus(o: Vec)  = Vec(x+o.x, y+o.y)
    operator fun times(n: Int) = Vec(x*n, y*n)
    operator fun unaryMinus()  = Vec(-x, -y)
    operator fun get(i: Int)   = if (i == 0) x else y
}
Vec(1,2) + Vec(3,4)                            // Vec(4,6)

# Operatorüberladung über Dunder-Methoden
class Vector:
    def __init__(self, x: float, y: float):
        self.x, self.y = x, y

    def __add__(self, other: "Vector") -> "Vector":
        return Vector(self.x + other.x, self.y + other.y)

    def __mul__(self, n: float) -> "Vector":
        return Vector(self.x * n, self.y * n)

    def __rmul__(self, n: float) -> "Vector":   # 3 * v
        return self.__mul__(n)

    def __neg__(self) -> "Vector":
        return Vector(-self.x, -self.y)

    def __abs__(self) -> float:
        return (self.x**2 + self.y**2) ** 0.5

    def __getitem__(self, i: int) -> float:
        return [self.x, self.y][i]

    def __iter__(self):
        yield self.x; yield self.y

    def __repr__(self) -> str:
        return f"Vector({self.x}, {self.y})"

v1 = Vector(1, 2)
v2 = Vector(3, 4)
v1 + v2                                        # Vector(4, 6)
v1 * 3; 3 * v1                                # Vector(3, 6)
abs(v1)                                        # 2.236...
x, y = v1                                      # Entpacken über __iter__

// ──────────────────────────────────────────────
// Alle Go-Operatoren
// ──────────────────────────────────────────────
func operators() {
    // Arithmetik
    a, b := 10, 3
    _ = a + b     // 13
    _ = a - b     // 7
    _ = a * b     // 30
    _ = a / b     // 3 (integer division!)
    _ = a % b     // 1
    _ = -a        // -10

    // Vergleich (Ergebnis bool)
    _ = a == b    // false
    _ = a != b    // true
    _ = a < b     // false
    _ = a <= b    // false
    _ = a > b     // true
    _ = a >= b    // true

    // Logisch
    x, y := true, false
    _ = x && y    // false (Short-Circuit: y wird nicht ausgewertet, wenn x=false)
    _ = x || y    // true  (Short-Circuit: y wird nicht ausgewertet, wenn x=true)
    _ = !x        // false

    // Bitweise
    n := 0b1010
    _ = n & 0b1100  // AND: 0b1000
    _ = n | 0b0101  // OR:  0b1111
    _ = n ^ 0b1100  // XOR: 0b0110
    _ = n << 2      // Left Shift: 0b101000
    _ = n >> 1      // Right Shift: 0b0101
    _ = ^n          // NOT: alle Bits invertieren
    _ = n &^ 0b1100 // AND NOT (Bit löschen): 0b0010

    // Zuweisung
    c := 5
    c += 3          // c = 8
    c -= 1          // c = 7
    c *= 2          // c = 14
    c /= 2          // c = 7
    c %= 3          // c = 1
    c <<= 1         // c = 2
    c >>= 1         // c = 1
    c &= 0xFF
    c |= 0b0001
    c ^= 0b0011

    // Inkrement / Dekrement — NUR Statement (kein Expression!)
    i := 0
    i++             // i = 1  (kein ++i, kein x = i++)
    i--             // i = 0

    _ = a; _ = b; _ = x; _ = y; _ = n; _ = c; _ = i
}

// cmp-Paket (Go 1.21+) — typsicherer Vergleich
// ──────────────────────────────────────────────
func cmpExamples() {
    cmp.Compare(1, 2)         // -1 (less)
    cmp.Compare(2, 2)         //  0 (equal)
    cmp.Compare(3, 2)         //  1 (greater)
    cmp.Compare("apple", "banana")  // -1

    // Wird in slices.SortFunc verwendet
    type Person struct{ Name string; Age int }
    people := []Person{{"Bob", 25}, {"Alice", 30}, {"Charlie", 25}}

    slices.SortFunc(people, func(a, b Person) int {
        // Nach Alter, dann nach Name sortieren
        if n := cmp.Compare(a.Age, b.Age); n != 0 { return n }
        return cmp.Compare(a.Name, b.Name)
    })
}

func pointerOps() {
    val := 42
    ptr := &val       // *int — Zeiger auf val
    *ptr = 100        // Wert über den Zeiger ändern
    fmt.Println(val)  // 100

    // Zeiger auf ein Struct-Feld
    type Point struct{ X, Y int }
    p := Point{1, 2}
    xPtr := &p.X
    *xPtr = 10
    fmt.Println(p)    // {10 2}
}

// min / max (Go 1.21+ eingebaute Funktionen!)
// ──────────────────────────────────────────────
func builtins() {
    // min und max — eingebaut (nicht aus dem math-Paket)
    min(3, 5, 1, 4, 2)    // 1 (variadic!)
    max(3, 5, 1, 4, 2)    // 5

    // clear — Slice oder Map leeren (Go 1.21+)
    s := []int{1, 2, 3}
    clear(s)               // alle Elemente = Zero Value: [0 0 0], len bleibt erhalten
    m := map[string]int{"a": 1}
    clear(m)               // alle Schlüssel entfernen

    // Eingebaute Funktionen (alle)
    // len, cap    — Länge/Kapazität
    // make        — Slice/Map/Channel erstellen
    // new         — Speicher reservieren, *T zurückgeben
    // append      — an Slice anhängen
    // copy        — Slice kopieren
    // delete      — aus Map löschen
    // close       — Channel schließen
    // panic       — Panic
    // recover     — Panic abfangen
    // print, println — eingebaute Ausgabe (für Debugging, nicht formatiert)
    // complex, real, imag — für komplexe Zahlen
    // min, max    — Go 1.21+
    // clear       — Go 1.21+

    _ = s; _ = m
}

// Ternärer Operator — GIBT ES NICHT in Go
// Verwenden Sie if-else oder eine Hilfsfunktion
// ──────────────────────────────────────────────
func ternary[T any](condition bool, ifTrue, ifFalse T) T {
    if condition { return ifTrue }
    return ifFalse
}
