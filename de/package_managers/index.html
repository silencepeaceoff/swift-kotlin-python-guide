    📦 Paketmanager und Build🔗

    // Swift Package Manager (SPM)
// Konfiguration: Package.swift

// swift-tools-version: 6.0
import PackageDescription

let package = Package(
    name: "MyApp",
    platforms: [.macOS(.v14), .iOS(.v17)],
    products: [
        .executable(name: "MyApp", targets: ["MyApp"]),
        .library(name: "MyLib", targets: ["MyLib"]),
    ],
    dependencies: [
        .package(url: "https://github.com/apple/swift-argument-parser", from: "1.3.0"),
        .package(url: "https://github.com/vapor/vapor", from: "4.89.0"),
    ],
    targets: [
        .executableTarget(name: "MyApp", dependencies: [
            "MyLib",
            .product(name: "ArgumentParser", package: "swift-argument-parser"),
        ]),
        .target(name: "MyLib"),
        .testTarget(name: "MyLibTests", dependencies: ["MyLib"]),
    ]
)

// Befehle:
// swift build                        — Build
// swift run                          — Start
// swift test                         — Tests
// swift package resolve              — Abhängigkeiten auflösen
// swift package update               — auf neuere Versionen aktualisieren

    // Gradle Kotlin DSL
// build.gradle.kts

plugins {
    kotlin("jvm") version "2.2.0"
    kotlin("plugin.serialization") version "2.2.0"
    application
}

group = "com.example"
version = "1.0.0"

repositories {
    mavenCentral()
    google()
}

dependencies {
    implementation("org.jetbrains.kotlinx:kotlinx-coroutines-core:1.9.0")
    implementation("org.jetbrains.kotlinx:kotlinx-serialization-json:1.7.3")
    implementation("io.ktor:ktor-client-cio:3.0.0")

    testImplementation(kotlin("test"))
    testImplementation("org.junit.jupiter:junit-jupiter:5.10.2")
    testImplementation("io.mockk:mockk:1.13.10")
}

application {
    mainClass.set("com.example.MainKt")
}

tasks.test { useJUnitPlatform() }

kotlin { jvmToolchain(21) }

// Befehle:
// ./gradlew build           — Build
// ./gradlew run             — Start
// ./gradlew test            — Tests
// ./gradlew dependencies    — Abhängigkeitsbaum

    # uv — blitzschneller Paketmanager (empfohlen 2026)
# curl -LsSf https://astral.sh/uv/install.sh | sh

# Projekt erstellen
# uv init myproject
# cd myproject

# pyproject.toml (einheitliche PEP-621-Konfiguration)
"""
[project]
name = "myproject"
version = "1.0.0"
requires-python = ">=3.13"
dependencies = [
    "httpx>=0.27",
    "pydantic>=2.0",
    "sqlalchemy[asyncio]>=2.0",
]

[project.optional-dependencies]
dev = [
    "pytest>=8.0",
    "pytest-asyncio>=0.23",
    "mypy>=1.10",
    "ruff>=0.4",
]

[tool.ruff]
target-version = "py313"
line-length = 100

[tool.pytest.ini_options]
asyncio_mode = "auto"

[tool.mypy]
strict = true
python_version = "3.13"
"""

# uv-Befehle:
# uv sync                    — Abhängigkeiten installieren
# uv add httpx               — Paket hinzufügen
# uv add --dev pytest        — Dev-Abhängigkeit hinzufügen
# uv remove httpx            — entfernen
# uv run python main.py      — Start
# uv run pytest              — Tests
# uv lock                    — Lock-Datei aktualisieren
# uv venv                    — Virtualenv erstellen
