    🌐 Kotlin Multiplatform (KMP)🔗

    Nur Kotlin

    KMP ist eine einzigartige Kotlin-Funktion für die plattformübergreifende Entwicklung: eine Codebasis für iOS, Android, Desktop und Web.

    // ── Projektstruktur ──
// my-project/
// ├── shared/                    ← gemeinsames KMP-Modul
// │   ├── src/
// │   │   ├── commonMain/        ← gemeinsamer Code (alle Plattformen)
// │   │   ├── commonTest/
// │   │   ├── androidMain/       ← Android-spezifisch
// │   │   ├── iosMain/           ← iOS-spezifisch
// │   │   └── desktopMain/       ← Desktop-spezifisch
// ├── androidApp/                ← Android-Anwendung
// ├── iosApp/                    ← iOS-Anwendung (Xcode)
// └── desktopApp/                ← Desktop-Anwendung

// shared/build.gradle.kts
kotlin {
    androidTarget { compilations.all { kotlinOptions { jvmTarget = "17" } } }
    iosX64(); iosArm64(); iosSimulatorArm64()
    jvm("desktop")

    sourceSets {
        commonMain.dependencies {
            implementation("org.jetbrains.kotlinx:kotlinx-coroutines-core:1.9.0")
            implementation("org.jetbrains.kotlinx:kotlinx-serialization-json:1.7.3")
            implementation("io.ktor:ktor-client-core:3.0.0")
        }
        androidMain.dependencies {
            implementation("io.ktor:ktor-client-android:3.0.0")
        }
        iosMain.dependencies {
            implementation("io.ktor:ktor-client-darwin:3.0.0")
        }
    }
}

// ── expect / actual — plattformabhängiger Code ──

// commonMain: den „Vertrag“ deklarieren
expect class PlatformInfo() {
    val name: String
    val version: String
}

expect fun currentTimeMillis(): Long

// androidMain: Implementierung für Android
actual class PlatformInfo actual constructor() {
    actual val name = "Android"
    actual val version = android.os.Build.VERSION.RELEASE
}

actual fun currentTimeMillis(): Long = System.currentTimeMillis()

// iosMain: Implementierung für iOS
actual class PlatformInfo actual constructor() {
    actual val name = "iOS"
    actual val version = UIDevice.currentDevice.systemVersion
}

actual fun currentTimeMillis(): Long =
    (NSDate().timeIntervalSince1970 * 1000).toLong()

// ── Gemeinsamer Code (commonMain) ──

interface UserRepository {
    suspend fun getUsers(): List<User>
    suspend fun getUserById(id: Int): User?
}

class UserViewModel(private val repo: UserRepository) {
    private val _state = MutableStateFlow<UserState>(UserState.Loading)
    val state: StateFlow<UserState> = _state.asStateFlow()

    fun loadUsers() {
        viewModelScope.launch {
            _state.value = try {
                UserState.Success(repo.getUsers())
            } catch (e: Exception) {
                UserState.Error(e.message ?: "Unknown error")
            }
        }
    }
}

sealed class UserState {
    data object Loading : UserState()
    data class Success(val users: List<User>) : UserState()
    data class Error(val message: String) : UserState()
}
