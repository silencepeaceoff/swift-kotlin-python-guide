@propertyWrapper
struct Clamped<T: Comparable> {
    private var value: T
    let range: ClosedRange<T>

    var wrappedValue: T {
        get { value }
        set { value = min(max(newValue, range.lowerBound), range.upperBound) }
    }

    var projectedValue: ClosedRange<T> { range }

    init(wrappedValue: T, _ range: ClosedRange<T>) {
        self.range = range
        self.value = min(max(wrappedValue, range.lowerBound), range.upperBound)
    }
}

struct Player {
    @Clamped(0...100) var health: Int = 100
    @Clamped(0...10)  var speed: Double = 5.0
}

p.health = 150          // automatisch 100
p.health = -10          // automatisch 0

// Eigene Delegate-Implementierung
class Clamped<T : Comparable<T>>(
    private var value: T,
    private val range: ClosedRange<T>
) : ReadWriteProperty<Any?, T> {
    override fun getValue(thisRef: Any?, property: KProperty<*>): T = value
    override fun setValue(thisRef: Any?, property: KProperty<*>, value: T) {
        this.value = value.coerceIn(range)
    }
}

fun <T : Comparable<T>> clamped(initial: T, range: ClosedRange<T>) =
    Clamped(initial, range)

class Player {
    var health: Int by clamped(100, 0..100)
    var speed: Double by clamped(5.0, 0.0..10.0)
}

    # Deskriptoren (__get__ / __set__ / __delete__)
class Clamped:
    """Deskriptor: klemmt den Wert in den Bereich [min_val, max_val]"""
    def __set_name__(self, owner, name: str) -> None:
        self.name = name
        self.private_name = f"_{name}"

    def __init__(self, min_val, max_val):
        self.min_val = min_val
        self.max_val = max_val

    def __get__(self, obj, objtype=None):
        if obj is None: return self
        return getattr(obj, self.private_name, self.min_val)

    def __set__(self, obj, value):
        clamped = max(self.min_val, min(self.max_val, value))
        setattr(obj, self.private_name, clamped)

# @property — praktische Syntax für Deskriptoren
class Temperature:
    def __init__(self, celsius: float = 0.0) -> None:
        self._celsius = celsius

    @property
    def celsius(self) -> float:
        return self._celsius

    @celsius.setter
    def celsius(self, value: float) -> None:
        if value < -273.15:
            raise ValueError("Below absolute zero!")
        self._celsius = value

    @property
    def fahrenheit(self) -> float:
        return self._celsius * 9/5 + 32

// ──────────────────────────────────────────────
// ① Clamped — Bereichsbegrenzung über Generics
// ──────────────────────────────────────────────
type Clamped[T cmp.Ordered] struct {
    value T
    min   T
    max   T
}

func NewClamped[T cmp.Ordered](value, min, max T) *Clamped[T] {
    return &Clamped[T]{value: clamp(value, min, max), min: min, max: max}
}

func clamp[T cmp.Ordered](v, lo, hi T) T {
    if v < lo { return lo }
    if v > hi { return hi }
    return v
}

func (c *Clamped[T]) Set(v T) { c.value = clamp(v, c.min, c.max) }
func (c *Clamped[T]) Get() T  { return c.value }

// Verwendung
type Player struct {
    Health *Clamped[int]
    Speed  *Clamped[float64]
}

func NewPlayer() *Player {
    return &Player{
        Health: NewClamped(100, 0, 100),
        Speed:  NewClamped(5.0, 0.0, 10.0),
    }
}

p := NewPlayer()
p.Health.Set(150)              // automatisch 100
p.Health.Get()                 // 100
p.Health.Set(-10)              // automatisch 0
p.Health.Get()                 // 0

// ──────────────────────────────────────────────
// ② Lazy[T] — verzögerte Initialisierung
// ──────────────────────────────────────────────
type Lazy[T any] struct {
    once  sync.Once
    value T
    init  func() T
}

func NewLazy[T any](init func() T) *Lazy[T] {
    return &Lazy[T]{init: init}
}

func (l *Lazy[T]) Get() T {
    l.once.Do(func() { l.value = l.init() })
    return l.value
}

// Verwendung
type DataManager struct {
    heavyData *Lazy[[]int]
}

func NewDataManager() *DataManager {
    return &DataManager{
        heavyData: NewLazy(func() []int {
            fmt.Println("Loading expensive data...")
            return make([]int, 10_000)
        }),
    }
}

dm := NewDataManager()
dm.heavyData.Get()             // "Loading expensive data..."
dm.heavyData.Get()             // aus dem Cache, ohne Ausgabe

// ──────────────────────────────────────────────
// ③ Observable[T] — Benachrichtigung bei Änderungen
// ──────────────────────────────────────────────
type Observable[T any] struct {
    mu       sync.RWMutex
    value    T
    handlers []func(T)
}

func NewObservable[T any](initial T) *Observable[T] {
    return &Observable[T]{value: initial}
}

func (o *Observable[T]) Get() T {
    o.mu.RLock(); defer o.mu.RUnlock()
    return o.value
}

func (o *Observable[T]) Set(v T) {
    o.mu.Lock()
    o.value = v
    handlers := o.handlers
    o.mu.Unlock()
    for _, h := range handlers { h(v) }       // außerhalb der Sperre benachrichtigen
}

func (o *Observable[T]) Subscribe(fn func(T)) func() {
    o.mu.Lock()
    o.handlers = append(o.handlers, fn)
    idx := len(o.handlers) - 1
    o.mu.Unlock()
    return func() {                            // gibt eine Abmeldefunktion zurück
        o.mu.Lock(); defer o.mu.Unlock()
        o.handlers = slices.Delete(o.handlers, idx, idx+1)
    }
}

// Verwendung
counter := NewObservable(0)
unsubscribe := counter.Subscribe(func(v int) {
    fmt.Printf("counter changed: %d\n", v)
})
counter.Set(1)                 // "counter changed: 1"
counter.Set(2)                 // "counter changed: 2"
unsubscribe()                  // abmelden
counter.Set(3)                 // keine Ausgabe

// ──────────────────────────────────────────────
// ④ Validated[T] — mit Validierung beim Schreiben
// ──────────────────────────────────────────────
type Validated[T any] struct {
    value     T
    validator func(T) error
}

func NewValidated[T any](initial T, validator func(T) error) (*Validated[T], error) {
    if err := validator(initial); err != nil { return nil, err }
    return &Validated[T]{value: initial, validator: validator}, nil
}

func (v *Validated[T]) Set(val T) error {
    if err := v.validator(val); err != nil { return err }
    v.value = val
    return nil
}

func (v *Validated[T]) Get() T { return v.value }

// Verwendung
age, _ := NewValidated(25, func(v int) error {
    if v < 0 || v > 150 { return fmt.Errorf("age %d out of range [0..150]", v) }
    return nil
})
age.Set(200)                   // error: "age 200 out of range"
age.Get()                      // 25 — unverändert
